\section{UMMIC: Universal Minimal Machine Information Coordinates}

\begin{definition}[Kolmogorov Complexity]
For string $x$, \textbf{Kolmogorov complexity} $K(x)$ minimal program length generating $x$ on universal Turing machine $U$:

$$
K(x):=\min\{\,|p|:\ U(p)=x\,\}.
$$
\end{definition}

\begin{definition}[UMMIC Coordinates]
On computation space $\mathcal{C}$, define \textbf{UMMIC coordinate system}:

$$
\xi_{\rm UMMIC}(x):=\bigl(K(x),\,I(x:y),\,H_{\rm alg}(x)\bigr),
$$

where $K(x)$ Kolmogorov complexity, $I(x:y)$ algorithmic mutual information, $H_{\rm alg}(x)$ algorithmic entropy rate.
\end{definition}

\subsection{Universal Properties}

\begin{theorem}[UMMIC Universality]
UMMIC coordinates universal up to additive constant: for any other coordinate system $\xi'$, exists constant $C$ such that

$$
|\xi_{\rm UMMIC}(x)-\xi'(x)|\le C
$$

for all $x\in\mathcal{C}$.
\end{theorem}

\section{UMMIC--WSIG Categorical Equivalence}

\begin{theorem}[Coordinate Isomorphism]
\label{thm:ummic-wsig-isomorphism}
Exists natural isomorphism $\Phi:\text{UMMIC}\to\text{WSIG}$ via:

\begin{align*}
K(x) &\mapsto \int_0^{E_x}\rho_{\rm rel}(E)\,dE\quad\text{(cumulative spectral density)}\\
I(x:y) &\mapsto \mathcal{N}_w^{(12)}[E_x,E_y]\quad\text{(cross-windowed readout)}\\
H_{\rm alg}(x) &\mapsto \lim_{h\downarrow 0}H^\natural[\rho_{w,h}]\quad\text{(scale-refined entropy)}
\end{align*}

preserving information-theoretic operations and categorical structure.
\end{theorem}

\begin{proof}[Proof Sketch]
Kolmogorov complexity as cumulative information maps to integrated spectral density via Birman--Kreĭn formula. Mutual information corresponds to joint windowed measurement. Algorithmic entropy rate matches refined windowed entropy via trinity scale.

Functoriality follows from:
\begin{itemize}
\item Composition: $(f\circ g)^*=g^*\circ f^*$ for both UMMIC and WSIG
\item Identity: $\text{id}^*=\text{id}$ preserved under $\Phi$
\item Additivity: $K(x,y)\approx K(x)+K(y|x)$ corresponds to $\rho_{\rm rel}$ additivity for product systems
\end{itemize}
\end{proof}
