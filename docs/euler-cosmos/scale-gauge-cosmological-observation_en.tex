\begin{abstract}
\textbf{Abstract:} We establish a cosmological observation theory centered on the duality between the scale gauge (denoted $a(t)$) and the internal observational metric (c-lock, denoted $R(t)$). Setting $a(t) = R(t)^{-1}$ and $\kappa(t) := \dot{a}/a = -\dot{R}/R$, cosmological redshift and time dilation unify as Mellin dilation on the energy axis: $1+z = a(t_0)/a(t_e) = R(t_e)/R(t_0)$, $\nu_0 = \nu_e/(1+z)$, $\Delta t_0 = (1+z)\Delta t_e$. All readouts are aligned on the ``mother scale'':
$$
\boxed{\,\rho(E) = -\xi'(E) = \frac{1}{2\pi}\operatorname{tr}\mathsf{Q}(E) = \frac{1}{2\pi i}\,\partial_E\log\det S(E) = \frac{\varphi'(E)}{2\pi}\,,\quad \mathsf{Q} := -i\,S^\dagger S'\,},
$$
where $\rho$ is the relative density of states, $\mathsf{Q}$ is the Wigner--Smith group delay matrix, and $\varphi$ is the total scattering phase. The mother scale is equivalently characterized by the Birman--Kreĭn formula and the group delay definition, providing cross-device comparable calibration unity (fixed units; dimension $E^{-1}$). Readout errors obey ``Nyquist--Poisson--Euler--Maclaurin (NPE) finite-order closure'': Nyquist cutoff eliminates aliasing, Poisson summation bridges discrete--continuous, and finite-order Euler--Maclaurin (EM) encapsulates endpoint errors with Bernoulli layers and tail bounds. Under Landau density threshold and Wexler--Raz dual frame conditions, window shrinking $r\downarrow$ maintains \textbf{non-increasing singularity}; whereas the monotonicity and scaling of Fisher information with respect to $r$ \textbf{depend on noise model and normalization choice}, and no universal monotonicity or $r^{-2}$ lower bound exists in general. The linear stabilizer preserving ``light cone + mother scale'' uniquely corresponds to the Lorentz group; in the FRW background, the unified frequency shift law $1+z = ((k_\mu u^\mu)_e)/((k_\mu u^\mu)_o)$ naturally yields Etherington's distance duality $D_L = (1+z)^2 D_A$ and Tolman's surface brightness $(1+z)^{-4}$. Via reversible causal automaton (RCA/QCA) semantics, we provide a unified formulation of ``light'', ``redshift'', ``shared master frequency'', and ``$c$-limited allocation of attention/action resources''. This paper distinguishes ``gauge'' from ``true dynamics'' and proposes falsifiability criteria and an engineering ``resolution allocation matrix'' approach.
\end{abstract}

\subsection{NPE Finite-Order Closure (Non-Asymptotic)}

\begin{itemize}
\item \textbf{Nyquist:} For band-limited targets, aliasing is zero when sampling rate exceeds twice the bandwidth; for non-band-limited cases, aliasing terms are explicitly accounted for \cite{nyquist}.
\item \textbf{Poisson:} Discrete--continuous bridging via Poisson summation, allowing lattice sums to switch to frequency-domain comb spectra \cite{dlmf_fourier}.
\item \textbf{Euler--Maclaurin (finite-order):} Endpoint layers and tail bounds given by Bernoulli polynomials, with truncation order $p$ fixed and error bounds auditable \cite{dlmf_bernoulli}.
\end{itemize}

\textbf{Error notation definition:} Denote by $\varepsilon_{\mathrm{alias}}(r) \ge 0$ the $L^1$-upper bound of \textbf{aliasing terms} introduced via Poisson summation; when the Nyquist condition is satisfied, $\varepsilon_{\mathrm{alias}}(r) = 0$. Denote by $\varepsilon_{\mathrm{EM}}(\delta;r;p) \ge 0$ the ``endpoint layer + tail'' budget after truncating the Euler--Maclaurin formula at fixed order $p$ ($\delta$ is the sampling step/equivalent grid spacing). There exists a constant $C_{2p}(r)$ such that
$$
\varepsilon_{\mathrm{EM}}(\delta;r;p) \le C_{2p}(r)\,\delta^{2p}.
$$
In general, \textbf{no universal power-law relation} with $r$ is claimed; if further regularity such as $w \in W^{2p,1}$, $h*\rho \in L^1_{\mathrm{loc}}$ is given, one may derive $C_{2p}(r) \lesssim r^{-2p}$, whence $\varepsilon_{\mathrm{EM}}(\delta;r;p) = O(\delta^{2p}r^{-2p})$ as a \textbf{model-dependent} conclusion. This order bound requires assuming $h*\rho$ is piecewise $C^{2p}$ on the thickening $K^\uparrow$ (where $K^\uparrow := K + \operatorname{supp}(w_r*h)$) of the working compact domain $K$ with bounded derivatives; \textbf{or} under weaker $BV$ assumptions, all jump contributions are incorporated into the Bernoulli endpoint layers before estimation (yielding only a $BV$-version bound, \textbf{not equivalent to piecewise $C^{2p}$}). Beyond this regularity regime, this paper does not claim such order.

\begin{theorem}[Redshift--Dilation Equivalence]\label{thm:redshift}
For the same photon observed at emission $t_e$ and observation $t_0$, we have
$$
1+z = \frac{a(t_0)}{a(t_e)} = \frac{R(t_e)}{R(t_0)},\qquad
\nu_0 = \frac{\nu_e}{1+z},\qquad
\Delta t_0 = (1+z)\Delta t_e.

\end{theorem}

\begin{proof}[Proof sketch]
In the FRW metric, the frequency is $\omega = -k_\mu u^\mu$, hence $1+z = ((k_\mu u^\mu)_e)/((k_\mu u^\mu)_o)$; parallel transport of $k^\mu$ along null geodesics yields $\omega \propto a^{-1}$. Substituting $a = R^{-1}$ completes the proof \cite{distance_cosmo}.
\end{proof}

\section{Essence of Resolution Enhancement: Information Geometry and Singularity Conservation}

Let the \textbf{normalized window} $w_r(x) := \frac{1}{r}w(x/r)$, where $w \ge 0$, $w \in W^{1,1}(\mathbb{R})$, $\int_{\mathbb{R}} w = 1$ (optionally: $\int x\,w(x)\,dx = 0$), convolution kernel $h$, and observable
$$
g_r(E) = (w_r * h * \rho)(E).
$$

\begin{proposition}[Scale Bound and Convergence of Gradient Response]\label{prop:gradient_scale}
Let $w \ge 0$, $w \in W^{1,1}(\mathbb{R})$, $\int w = 1$, and \textbf{$h*\rho \in L^1_{\mathrm{loc}}$ (or $BV$)}, with $g_r = w_r * h * \rho$. For compact domain $K$,
$$
|\partial_E g_r|_{L^1(K)} \le \frac{|w'|_{L^1}}{r}\,|h*\rho|_{L^1(K^\uparrow)}.
$$

\textbf{Convergence by cases:}

(i) If $h*\rho \in W^{1,1}(K^\uparrow)$ and $w \ge 0$, $\int w = 1$, then
$$
\lim_{r\downarrow 0}\,|\partial_E g_r - (h*\rho)'|_{L^1(K)} = 0,\qquad
|\partial_E g_r|_{L^1(K)} \le |(h*\rho)'|_{L^1(K^\uparrow)}.
$$

(ii) If $h*\rho \in BV(K^\uparrow)$ (not necessarily in $W^{1,1}$), then
$$
g_r \xrightarrow[r\downarrow 0]{} h*\rho\quad\text{in }L^1(K),\qquad
|\partial_E g_r|_{L^1(K)} \le \mathrm{TV}(h*\rho;K^\uparrow),
$$
and $\partial_E g_r \stackrel{*}{\rightharpoonup} D(h*\rho)$ in the weak$^*$ sense in measure space. We do not claim convergence of $|\partial_E g_r - (h*\rho)'|_{L^1}$ in this case.

\textbf{NPE estimator version} (for discrete implementation $\widehat{g}_r$):
$$
|\partial_E \widehat{g}_r|_{L^1(K)} \le \frac{|w'|_{L^1}}{r}\,|h*\rho|_{L^1(K^\uparrow)} + \varepsilon_{\mathrm{alias}}(r) + \varepsilon_{\mathrm{EM}}(\delta;r;p).
$$
If the Nyquist condition is satisfied, $\varepsilon_{\mathrm{alias}}(r) = 0$, leaving only the EM endpoint--tail budget.
\end{proposition}

The above shows: reducing $r$ improves edge \textbf{approximation}, but does not produce a universal $1/r$ \textbf{lower bound} growth. Here $K^\uparrow := K + \operatorname{supp}(w_r * h)$ denotes the thickening of the compact domain $K$ by the effective support of the convolution kernel \cite{dlmf_bernoulli}.

\begin{proposition}[Model Dependence of Fisher Information]\label{prop:fisher_model}
Under the premise that Nyquist and NPE error budgets ($\varepsilon_{\mathrm{alias}}$, $\varepsilon_{\mathrm{EM}}$) are auditable, the monotonicity and scaling of $\mathcal{I}_r(\theta)$ with respect to $r$ \textbf{depend on noise model and normalization choice}; in general, \textbf{no universal $r^{-2}$ lower bound or monotonicity conclusion exists}. Once noise statistics (e.g., AWGN/Poisson) and window normalization (e.g., $\int w = 1$ or $|w_r|_2$ fixed) are specified, one may derive the corresponding $r$-scaling and comparison results \cite{nyquist}.
\end{proposition}

\begin{theorem}[Non-Increasing Singularity]\label{thm:singularity}
Legitimate window switching ($w \mapsto w_r$ with fixed-order EM budget) corresponds to smoothing that does not \textbf{introduce new} singularities of $h*\rho$; thus under alias control and auditable EM error, resolution enhancement does not ``manufacture spurious peaks''. The \textbf{location and order} of singularities may be affected by smoothing; this paper makes no invariance claims \cite{dlmf_fourier}.
\end{theorem}

\section*{Proof Appendix (Selected)}

\subsection*{A. Birman--Kreĭn--Group Delay--Phase Derivative Trinity}

Let $S(E)$ be a unitary scattering matrix. By Smith's definition $\mathsf{Q}(E) = -i\,S^\dagger S'$,
$$
\operatorname{tr}\mathsf{Q} = -i\,\operatorname{tr}(S^\dagger S') = -i\,\partial_E\log\det S\,,
$$
using $\partial_E\log\det S = \operatorname{tr}(S^{-1}S') = \operatorname{tr}(S^\dagger S')$ (since $S$ is unitary). By $\det S(E) = e^{-2\pi i\xi(E)}$ and $\operatorname{tr}\mathsf{Q} = -i\,\partial_E\log\det S$, we get $\frac{1}{2\pi}\operatorname{tr}\mathsf{Q} = -\xi'(E)$. Also $\varphi(E) = \arg\det S(E) = -2\pi\xi(E)$, hence $\varphi'(E) = -2\pi\xi'(E)$. Thus
$$
\rho(E) = -\xi'(E) = \frac{1}{2\pi}\operatorname{tr}\mathsf{Q}(E) = \frac{1}{2\pi i}\partial_E\log\det S(E) = \frac{\varphi'(E)}{2\pi}\,.
$$
\cite{wigner2018}
