\begin{abstract}
We propose \textbf{Null--Modular double cover} carried by zero-measure boundaries of causal diamonds, decomposing modular Hamiltonians into local energy flux integrals on two null sheets in vacuum quadratic form sense. Through a \textbf{total-order approximation bridge lemma}, general diamonds are reduced to monotonic half-space family limits on the same zero-measure hyperplane, with quadratic form closedness and dominated convergence ensuring limit independence of approximation paths. We establish \textbf{modular Hamiltonian inclusion--exclusion identities} and \textbf{Markov splicing} for \textbf{overlapping causal diamond chains}; for non-totally-ordered cuts, we introduce \textbf{Markov gap line density} quantitatively characterizing failure with comparison inequalities versus stratification degree. On the scattering side, under \textbf{distributional Birman--Kreĭn--Friedel--Lloyd--Wigner--Smith calibration}, we introduce \textbf{windowed readout}, providing \textbf{visible constants and threshold inequalities} via Toeplitz/Berezin compression, Euler--Maclaurin and Poisson disciplines, thereby proving \textbf{chain $\mathbb{Z}_2$ parity threshold stability} with robustness conditions for \textbf{weakly non-unitary perturbations}. On the geometric side, \textbf{half-sided modular inclusion} constitutes one-parameter semigroup for chain advancement; in holographic limits, \textbf{JLMS equality} lifts boundary inclusion--exclusion--Markov to bulk entanglement wedge normal modular flow, with dimensional upper bounds for subleading $1/N$ corrections. Finally, we explicitly compute \textbf{GHY joint terms} and \textbf{$\mathbb{Z}_2$ ledger consistency} of square-root splicing classes in minimal models in $1+1$ and $2+1$ dimensions, providing \textbf{reproducible experimental parameter tables} and \textbf{verification checklists}.
\end{abstract}

\section{Model \& Assumptions}

\subsection{Quadratic Form Framework and Natural Domain}

Take Minkowski spacetime $\mathbb{R}^{1,d-1}$ ($d\ge2$). Let $\mathcal{D}_0$ be dense domain of energy-bounded vectors in vacuum.

\textbf{Notation and measure convention}: Zero-measure boundary decomposes into two sheets $\widetilde{E}=E^+\sqcup E^-$; notation $\int_{E^\sigma}(\cdots)\,d\lambda\,d^{d-2}x_\perp$ refers to standard measure integration on this sheet by affine parameter $\lambda$ and transverse coordinate $x_\perp$.

Assume for any region $R$ there exists lower bounded closed quadratic form

$$
\mathfrak{k}_R[\psi]
:=\sum_{\sigma=\pm}\int_{E^\sigma} g_\sigma^{(R)}(\lambda,x_\perp)\,
\langle\psi, T_{\sigma\sigma}(\lambda,x_\perp)\psi\rangle\,d\lambda\,d^{d-2}x_\perp,
\quad \psi\in\mathcal{D}_0,
$$

thus there exists self-adjoint operator $K_R$ satisfying $\langle\psi,K_R\psi\rangle=\mathfrak{k}_R[\psi]$. CFT's spherical regions/wedges and their conformal images yield exact geometric equalities.

Let $\mathfrak{k}_R$ have lower bound $a_R\in\mathbb{R}$, i.e., $\mathfrak{k}_R[\psi]\ge a_R|\psi|^2$. Take any $c_R>-a_R$, define \textbf{shifted graph norm}

$$|\psi|_{\mathfrak{k}_R,c_R}^2:=|\psi|^2+\big(\mathfrak{k}_R[\psi]+c_R|\psi|^2\big),$$

then $(\mathcal{D}(\mathfrak{k}_R),|\cdot|_{\mathfrak{k}_R,c_R})$ is complete, compatible with representation theorem for self-adjoint operator $K_R$.

\subsection{Zero-Measure Localization and QNEC}

In zero-measure half-space $R_V=\{u=0,\ v\ge V(x_\perp)\}$ ($V\in C^2$),

$$
K_V=2\pi\int d^{d-2}x_\perp\int_{V(x_\perp)}^\infty (v-V)\,T_{vv}(v,x_\perp)\,dv
$$

holds as quadratic form identity; its second-order variational kernel is $2\pi\,T_{vv}$, consistent with vacuum QNEC saturation.

\subsection{Scattering--Information Calibration and Windowing}

Unitary scattering matrix $S(E)$ piecewise $C^{2m}$ with $S(E)-\mathbb{I}$ trace-class within energy band; define

$$
Q(E):=-\mathrm{i}\,S^\dagger\partial_E S,\quad
\varphi(E):=\tfrac{1}{2}\arg\det S(E),\quad
\rho_{\rm rel}(E):=\tfrac{1}{2\pi}\operatorname{tr}Q(E).
$$

Employ window function $h\in\mathcal{S}(\mathbb{R})$ (e.g., Gaussian), \textbf{or} $h\in C_c^{2m+1}(\mathbb{R})$ with endpoint jets up to $2m$ order vanishing ($m\ge1$). In this case $\widehat{h}(\omega)=O(|\omega|^{-(2m+1)})$. If $h$ only piecewise $C^{2m}$ with compact support (endpoints allow corners, including Kaiser--Bessel), adopt \textbf{corner tail bound} (at least $O(|\omega|^{-2})$), whereby Theorem G's Poisson aliasing series converges. Corresponding Toeplitz/Berezin compression and trace formulas follow error decomposition in §3.5, where \textbf{endpoint remainder $R_{\rm EM}$}: for $C_c^\infty$ windows take $O(\ell^{-(m-1)})$; for piecewise $C^{2m}$ compact support windows (including Kaiser--Bessel) adopt \textbf{corner version} estimate (order generally drops to $O(\ell^{-1})$), incorporated into total error budget $\mathcal{E}_h(\gamma)$.

\textbf{Additional assumption (Toeplitz commutator integrability)}: On any examined energy band $\mathcal{I}$, $\partial_E S(E)\in\mathfrak{S}_2$ and $\displaystyle\int_{\mathcal{I}}|\partial_E S(E)|_2\,dE<\infty$. Thus $R_{\rm T}\le C_{\rm T}\,\ell^{-1/2}\!\int_{\mathcal{I}}|\partial_E S|_2\,dE$ is bounded.

\textbf{Global convention (window and tail term)}: Set $\int_{\mathbb{R}}h=1$ \textbf{and $h\ge0$}, scale $h_\ell(E)=\ell^{-1}h(E/\ell)$. Define

$$\boxed{R_{\rm tail}(\ell,\mathcal{I},E_0):=\int_{\mathbb{R}\setminus\mathcal{I}(\gamma)} |h_\ell(E-E_0)|\,dE\in[0,1].}$$

\emph{Note}: In this case $R_{\rm tail}=1-\int_{\mathcal{I}(\gamma)}h_\ell(E-E_0)\,dE$.

\textbf{Notation convention (Poisson step size)}: Denote $\Delta>0$ as energy band segmentation/frequency sampling step size (grid spacing); in Poisson resummation estimate take

$$\int_{\mathcal{I}}|R_{\rm P}|\,dE\ \le\ C_h\sum_{|q|\ge1}\big|\widehat{h}\left(2\pi q\,\ell/\Delta\right)\big|,$$

consistently using this $\Delta$ as in §3.5's identically named term.

\subsection{Double-Sheet Geometric Decomposition and Total-Order Approximation Bridge}

\begin{theorem}[A: Double-Sheet Geometric Decomposition]

$$
K_D = 2\pi\sum_{\sigma=\pm}\int_{E^\sigma} g_\sigma(\lambda,x_\perp)\,
T_{\sigma\sigma}(\lambda,x_\perp)\,d\lambda\,d^{d-2}x_\perp,
$$

where $T_{++}=T_{vv}$, $T_{--}=T_{uu}$. In CFT spherical diamonds $g_\sigma(\lambda)=\lambda(1-\lambda)$.

[\textbf{Quadratic form; domain: vacuum, CFT exact equality}]
\end{theorem}

\begin{assumption}[A$'$: Null Energy Flux Uniform Integrability]
For any $\psi\in\mathcal{D}_0$ and geometrically bounded monotonic approximation family $\{R_{V_\alpha}^\pm\}$, there exists $H_\sigma\in L^1_{\mathrm{loc}}(E^\sigma\times\mathbb{R}^{d-2})$ such that

$$
\big|g^{(\alpha)}_\sigma(\lambda,x_\perp)\,\langle\psi,T_{\sigma\sigma}(\lambda,x_\perp)\psi\rangle\big|
\le H_\sigma(\lambda,x_\perp)
$$

holds almost everywhere, and $\sup_\alpha\int_{\mathcal{K}} H_\sigma<\infty$ for any compact set $\mathcal{K}\subset E^\sigma\times\mathbb{R}^{d-2}$.
\end{assumption}

\begin{lemma}[A: Ordered Cut Approximation]
There exists monotonic half-space family $\{R_{V_\alpha}^\pm\}$ along $E^\pm$ such that

$$
\langle\psi,K_D\psi\rangle=\lim_{\alpha\to\infty}
\sum_{\sigma=\pm}2\pi\!\int_{E^\sigma}
g^{(\alpha)}_\sigma\,\langle\psi,T_{\sigma\sigma}\psi\rangle,
\qquad g^{(\alpha)}_\sigma\to g_\sigma\ \text{in}\ L^1_{\mathrm{loc}},
$$

and limit is independent of chosen ordered approximation.

[\textbf{Quadratic form convergence; domain: vacuum, vacuum QNEC saturation}]
\end{lemma}

\textbf{Exclusion remark}: Without BW/HSMI or boundary roughness breaking vacuum QNEC saturation, above decomposition may not hold.

\begin{assumption}[A$''$: Quadratic Form Lower Bound and Closedness Threshold]
Assume all participating regions $R$ have quadratic forms $\mathfrak{k}_R$ with uniform lower bound $a\in\mathbb{R}$, i.e., $\mathfrak{k}_R[\psi]\ge a\,|\psi|^2$. Take any $c>-a$ defining shifted graph norm $|\psi|_{\mathfrak{k}_R,c}^2=|\psi|^2+(\mathfrak{k}_R[\psi]+c|\psi|^2)$, then $\mathfrak{k}_R$ closed and $\mathcal{D}(\mathfrak{k}_R)$ complete under $|\cdot|_{\mathfrak{k}_R,c}$.
\end{assumption}

\begin{proposition}[A.1: Necessary and Sufficient Condition for Limit Path Independence]
Under Assumptions A$'$ and A$''$, if along any two monotonic approximation families $\{R_{V_\alpha}\}$, $\{R_{\widetilde{V}_\beta}\}$ we have $g^{(\alpha)}\to g$, $\widetilde{g}^{(\beta)}\to g$ in $L^1_{\mathrm{loc}}$, then for each $\psi\in\mathcal{D}_0$,

$$
\lim_{\alpha\to\infty}\!\sum_{\sigma}\!\int g^{(\alpha)}_\sigma\,\langle\psi,T_{\sigma\sigma}\psi\rangle
=\lim_{\beta\to\infty}\!\sum_{\sigma}\!\int \widetilde{g}^{(\beta)}_\sigma\,\langle\psi,T_{\sigma\sigma}\psi\rangle .
$$

Reason: Dominated convergence identifies each approximation's limit with $g$; closedness and lower bound yield quadratic form continuity, thus independent of approximation path.
\end{proposition}

\subsection{Inclusion--Exclusion and Closedness}

\begin{theorem}[B: Inclusion--Exclusion Identity]
For $\{R_{V_i}\}_{i=1}^N$ on same zero-measure surface,

$$
K_{\cup_i R_{V_i}}
=\sum_{k=1}^N(-1)^{k-1}\!\!\sum_{1\le i_1<\cdots<i_k\le N}\!
K_{R_{V_{i_1}}\cap\cdots\cap R_{V_{i_k}}}.
$$

Derived from pointwise identity $(v-\min_i V_i)_+ = \sum_{k\ge1}(-1)^{k-1}\sum_{|I|=k}(v-\max_{i\in I}V_i)_+$.

[\textbf{Quadratic form; domain: vacuum, $V_i$ piecewise smooth}]
\end{theorem}

\begin{proposition}[B: Closedness]
Denote $\mathfrak{k}:=\mathfrak{k}_{\cup_i R_{V_i}}$ as container domain closed quadratic form, with lower bound $a\in\mathbb{R}$. Take any $c>-a$. If

$$\psi_n,\psi\in \mathcal{D}(\mathfrak{k})\cap\bigcap_{I\ne\emptyset}\mathcal{D}(\mathfrak{k}_{R_{V_I}}),
\qquad \psi_n\to\psi\ \text{under shifted graph norm }|\cdot|_{\mathfrak{k},c},$$

then inclusion--exclusion identity's both sides for quadratic form values on $\psi_n$ converge simultaneously to values on $\psi$; thus identity closes on above form domain. Where

$$|\psi|_{\mathfrak{k},c}^2:=|\psi|^2+\big(\mathfrak{k}[\psi]+c|\psi|^2\big).$$

[\textbf{Quadratic form closedness}]
\end{proposition}

\textbf{Operational domain remark}: Above closedness holds on common form domain $\mathcal{D}_*:=\mathcal{D}(\mathfrak{k})\cap\bigcap_{I\ne\emptyset}\mathcal{D}(\mathfrak{k}_{R_{V_I}})$; for chain applications, taking $V_i$ piecewise $C^1$ with uniform Lipschitz constant ensures $\mathcal{D}_*$ non-empty and dense.

\subsection{Markov Splicing, Petz Recovery, and Non-Total-Order Gap}

\begin{theorem}[C: Markov Splicing]
Under same-surface total order, vacuum satisfies

$$
I(D_{j-1}:D_{j+1}\mid D_j)=0,\qquad
K_{D_{j-1}\cup D_j}+K_{D_j\cup D_{j+1}}-K_{D_j}-K_{D_{j-1}\cup D_j\cup D_{j+1}}=0 .
$$

[\textbf{Information equivalence; domain: vacuum, split/strong additivity}]
\end{theorem}

\begin{theorem}[C$'$: Markov Gap for Non-Total-Order]
\textbf{Definition (stratification degree)}: Let $V_i^\pm(x_\perp)$ be thresholds on $E^\pm$ respectively, define

$$\boxed{\kappa(x_\perp):=\#\{(a,b):a<b,\ (V_a^+-V_b^+)(V_a^--V_b^-)<0\}.}$$

\emph{Note}: Under total-order cut $\kappa\equiv 0$. Thus $\iota$ monotonically non-decreasing in $\kappa$ yields comparison inequality.

To bound $\iota(v,x_\perp)$'s $v$ domain, denote

$$
v_-(x_\perp):=\min_i V_i^+(x_\perp),\qquad v_+(x_\perp):=\max_i V_i^+(x_\perp),
$$

i.e., effective support interval endpoints covered by chain on $E^+$ sheet; below statements about $v$ understood within $[v_-(x_\perp),v_+(x_\perp)]$.

\textbf{Markov gap line density} $\iota(v,x_\perp)\ge0$ defined by relative entropy density kernel satisfies

$$
I(D_{j-1}\!:\!D_{j+1}\mid D_j)=\iint\iota(v,x_\perp)\,dv\,d^{d-2}x_\perp,
\qquad
\iota\ \text{monotonically non-decreasing in}\ \kappa.
$$

Particularly, under total order $\kappa\equiv0$ and $I(D_{j-1}\!:\!D_{j+1}\mid D_j)=0$ (Markov saturation).

[\textbf{Inequality; domain: vacuum}]
\end{theorem}

\begin{lemma}[C.1: Stratification Degree--Gap Comparison]
Assume $V_i^\pm$ piecewise $C^1$ with only finitely many crossings at each $x_\perp$. Then there exists constant $c_*>0$ (depending on $\sup|\partial V_i^\pm|$ and crossing number upper bound) such that in distributional sense

$$
\iota(v,x_\perp)\ \ge\ c_*\,\kappa(x_\perp)\,\mathbf{1}_{\{v\in[v_-(x_\perp),v_+(x_\perp)]\}} .
$$

Combined with Fawzi--Renner lower bound, yields quantitative gap lower bound under non-total-order.
\end{lemma}

\textbf{Fidelity convention}: This paper uniformly takes Uhlmann fidelity (not squared)

$$F(\rho,\sigma):=\big|\sqrt{\rho}\sqrt{\sigma}\big|_1\in[0,1].$$

Accordingly, Fawzi--Renner inequality writes

$$I(A:C\mid B)\ge -2\ln F,\qquad\text{equivalently }F\ge e^{-I(A:C\mid B)/2}.$$

\begin{theorem}[D: Petz Recovery and Stability — Self-Consistent Version]
Denote $A=D_{j-1}$, $B=D_j$, $C=D_{j+1}$. Take forgetting channel

$$\Phi_{BC\to B}(X_{BC})=\operatorname{Tr}_C[X_{BC}],\qquad \Phi^*(Y_B)=Y_B\otimes \mathbb{I}_C .$$

With $\sigma_{BC}=\rho_{BC}$ as reference state (thus $\sigma_B=\rho_B$), Petz recovery map $\mathcal{R}_{B\to BC}$ defined as

$$\boxed{\ \mathcal{R}_{B\to BC}(X_B)=\sigma_{BC}^{1/2}\big(\sigma_B^{-1/2}X_B\,\sigma_B^{-1/2}\otimes\mathbb{I}_C\big)\sigma_{BC}^{1/2}\ },$$

where inverse takes pseudo-inverse on $\mathrm{supp}(\sigma_B)$. \textbf{If and only if} $I(A:C\mid B)=0$ perfect recovery exists

$$(\mathrm{id}_A\otimes \mathcal{R}_{B\to BC})(\rho_{AB})=\rho_{ABC}.$$

Generally there exists \textbf{rotationally averaged Petz recovery} $\mathcal{R}^{\rm rot}_{B\to BC}$ such that

$$I(A:C\mid B)\ \ge\ -2\ln F\left(\rho_{ABC},\ (\mathrm{id}_A\otimes \mathcal{R}^{\rm rot}_{B\to BC})(\rho_{AB})\right),
\quad\text{equivalently }F\ge e^{-I(A:C\mid B)/2}.$$

Above inequality generally not guaranteed for unrotated $\mathcal{R}_{B\to BC}$; this paper uniformly adopts $\mathcal{R}^{\rm rot}_{B\to BC}$ for stability propositions.

[\textbf{Perfect recovery/stability; domain: Markov saturation}]
\end{theorem}

\subsection{Distributional KFL--WS Calibration and Windowed Parity Threshold}

\textbf{Non-smooth window transition and error incorporation}: If window $h\in C_c^0$ piecewise $C^{2m}$ within support (endpoints allow corners), take standard smoothing kernel $\rho_\delta$ and define $h_{\ell,\delta}:=h_\ell*\rho_\delta$. Then for each fixed $\ell>0$,

$$\big|h_{\ell,\delta}-h_\ell\big|_{L^1(\mathbb{R})}=O(\delta),$$

and Theorem F, Toeplitz/Berezin compression and trace formula first apply to $h_{\ell,\delta}$; by triangle inequality

$$R_{\rm smooth}(\delta):=\int_{\mathcal{I}(\gamma)}|h_{\ell,\delta}-h_\ell|\,dE$$

incorporates into total error budget $\mathcal{E}_h(\gamma)$. Under Theorem G threshold conditions, choose $\delta=\delta(\ell,m)$ making $R_{\rm smooth}(\delta)\le \frac{1}{2}\,\delta_*(\gamma)$, preserving same parity threshold conclusion as $h_\ell$.

\begin{theorem}[F: Distributional Calibration Identity]
For $h\in C_c^\infty(\mathbb{R})$ (or $h\in\mathcal{S}(\mathbb{R})$),

$$
\int \partial_E\arg\det S(E)\,h(E)\,dE
=\int \operatorname{tr}Q(E)\,h(E)\,dE
=-2\pi\!\int \xi'(E)\,h(E)\,dE ,
$$

where $\xi$ is spectral shift function. (Convention: Birman--Kreĭn takes $\det S(E)=e^{-2\pi \mathrm{i}\,\xi(E)}$.) Energy band thresholds and embedded eigenstates avoided by choosing $\operatorname{supp}h$; long-range potentials require corresponding generalized KFL.

[\textbf{Distributional equality; domain: $S-\mathbb{I}\in\mathfrak{S}_1$, piecewise smooth}]
\end{theorem}

\begin{proposition}[F$'$: Relative/Modified Calibration]
If $S_0(E)$ is reference scattering co-analytically segment-wise within energy band, without zeros/poles, and

$$
U(E):=S(E)S_0(E)^{-1},\qquad U(E)-\mathbb{I}\in\mathfrak{S}_2,\quad \partial_E U\in\mathfrak{S}_2,\ \int_{\mathcal{I}}|\partial_E U|_2<\infty,
$$

then Carleman determinant satisfies

$$
\int \partial_E\arg\det_2 U(E)\,h(E)\,dE
=\int \operatorname{tr}\big(Q(E)-Q_0(E)\big)\,h(E)\,dE,
$$

where $Q=-\mathrm{i}\,S^\dagger\partial_E S$, $Q_0=-\mathrm{i}\,S_0^\dagger\partial_E S_0$. If $S$ unitary and $S_0=\mathbb{I}$, above reduces to Theorem F. This proposition yields phase--group delay--spectral shift consistency under ``non-trace-class but relatively second-order traceable'' window.
\end{proposition}

\textbf{Note ($\pi/2$ buffer origin)}: In parity determination, $(-1)^{\lfloor\Theta/\pi\rfloor}$ only flips when $\Theta$ crosses odd multiples of $\pi$. Converging perturbation total to $<\pi/2$ ensures not crossing nearest integer multiple of $\pi$, thus consistent with unperturbed parity; taking $\delta_*(\gamma)=\min\{\pi/2,\delta_{\rm gap}(\gamma)\}-\varepsilon$ is explicit formulation of this buffer.

\textbf{Branch convention (arg regularization)}: Take continuous branch of $\arg\det S$ defined within energy band except countable discrete set; its distributional derivative $\partial_E\arg\det S$ independent of branch's $2\pi$ jump choice, as $h\in C_c^\infty$ annihilates jumps and matches $\operatorname{tr}Q$ via DOI/Helffer--Sjöstrand representation.

\begin{theorem}[G: Windowed Parity Threshold; With-Gap Threshold]
Let

$$
\Theta_h(\gamma):=\frac{1}{2}\int_{\mathcal{I}(\gamma)}\operatorname{tr}Q(E)\,h_\ell(E-E_0)\,dE,
\qquad \nu_{\rm chain}(\gamma):=(-1)^{\lfloor \Theta_h(\gamma)/\pi\rfloor}.
$$

Define unwindowed limit

$$
\Theta_{\rm geom}(\gamma):=\frac{1}{2}\int_{\mathcal{I}(\gamma)}\operatorname{tr}Q(E)\,dE
=\int_{\mathcal{I}(\gamma)}\varphi'(E)\,dE
=\varphi(E_2)-\varphi(E_1),
$$

where $\mathcal{I}(\gamma)=[E_1,E_2]$, $\varphi(E)=\tfrac{1}{2}\arg\det S(E)$. Define gap

$$
\delta_{\rm gap}(\gamma):=\operatorname{dist}\big(\Theta_{\rm geom}(\gamma),\pi\mathbb{Z}\big).
$$

Under $\int_{\mathbb{R}}h=1$, and setting

$$
\boxed{\ \mathcal{E}_h(\gamma)\ :=\ \underbrace{\int_{\mathcal{I}}|R_{\rm EM}|\,dE}_{\text{EM endpoint}}
+\underbrace{\int_{\mathcal{I}}|R_{\rm P}|\,dE}_{\text{Poisson aliasing}}
+\underbrace{C_{\rm T}\,\ell^{-1/2}\!\int_{\mathcal{I}}\!|\partial_E S|_2\,dE}_{\text{Toeplitz commutator}}
+\ \underbrace{R_{\rm tail}(\ell,\mathcal{I},E_0)}_{\text{out-of-interval tail}}\ \le\ \delta_*(\gamma)\ } ,
$$

where

$$R_{\rm tail}(\ell,\mathcal{I},E_0):=\int_{\mathbb{R}\setminus\mathcal{I}(\gamma)} |h_\ell(E-E_0)|\,dE\in[0,1].$$

\emph{Note}: If $h\ge0$ and $\int_{\mathbb{R}} h=1$, then $R_{\rm tail}=1-\int_{\mathcal{I}(\gamma)} h_\ell(E-E_0)\,dE$. Set $\delta_*(\gamma):=\min\big\{\tfrac{\pi}{2},\delta_{\rm gap}(\gamma)\big\}-\varepsilon$. If there exist $\ell>0$, $\Delta>0$, $m\in\mathbb{N}$ and $\varepsilon\in(0,\delta_{\rm gap}(\gamma))$ making above inequality hold, then for any window center $E_0$ satisfying above window quality conditions,

$$
\nu_{\rm chain}(\gamma)=(-1)^{\lfloor \Theta_h(\gamma)/\pi\rfloor}
=(-1)^{\lfloor \Theta_{\rm geom}(\gamma)/\pi\rfloor}.
$$

Here
$\bullet$ $R_{\rm EM}$ is Euler--Maclaurin endpoint remainder, satisfying $\int|R_{\rm EM}|\le C_m\,\ell^{-(m-1)}$;
$\bullet$ $R_{\rm P}$ is Poisson aliasing, satisfying

$$\int_{\mathcal{I}}|R_{\rm P}|\,dE\ \le\ C_h\sum_{|q|\ge1}\Bigl\lvert\widehat{h}\left(\tfrac{2\pi q\,\ell}{\Delta}\right)\Bigr\rvert,$$

where \textbf{$\Delta>0$} is \textbf{energy sampling step size (energy band lattice spacing)} used in Poisson summation;
$\bullet$ $R_{\rm T}$ is Toeplitz commutator term, under assumption $\partial_E S\in\mathfrak{S}_2$ and $\int_{\mathcal{I}}|\partial_E S|_2\,dE<\infty$ satisfying $R_{\rm T}\le C_{\rm T}\,\ell^{-1/2}\!\int_{\mathcal{I}}|\partial_E S|_2\,dE$;
$\bullet$ \textbf{Out-of-interval tail term}:

$$R_{\rm tail}(\ell,\mathcal{I},E_0):=\int_{\mathbb{R}\setminus\mathcal{I}(\gamma)} |h_\ell(E-E_0)|\,dE\in[0,1].$$

\emph{Note}: If $h\ge0$ and $\int_{\mathbb{R}} h=1$, then $R_{\rm tail}=1-\int_{\mathcal{I}(\gamma)} h_\ell(E-E_0)\,dE$.

\textbf{Note}: For piecewise smooth compact support windows (e.g., Kaiser), above $R_{\rm EM}$'s $C_m\,\ell^{-(m-1)}$ should be replaced with corner estimate (e.g., $O(\ell^{-1})$), other three terms $R_{\rm P},R_{\rm T},R_{\rm tail}$ unchanged. By above decay orders, $\displaystyle R_{\rm P}\le C_h\sum_{|q|\ge1}\big|\widehat{h}\left(2\pi q\,\ell/\Delta\right)\big|$ \textbf{finite}, maintaining same order as corner estimate.

[\textbf{Windowed distributional equality + explicit threshold; domain: unitary scattering, $h\in C_c^\infty$ or $h\in\mathcal{S}$}]
\end{theorem}

\begin{lemma}[T: Toeplitz/Berezin Compression Error]
Let $\mathsf{T}_\ell$ be windowed compression operator on energy axis (kernel is convolution with $h_\ell(E-E')$), let $Q(E)=-\mathrm{i}\,S(E)^\dagger\partial_E S(E)$, with $\partial_E S\in\mathfrak{S}_2$ satisfying $\int_{\mathcal{I}}|\partial_E S|_2\,dE<\infty$. Then there exists constant $C_T>0$ such that

$$
\Big\lvert\operatorname{tr}\big(Q*h_\ell\big)-\int Q(E)\,h_\ell(E-E_0)\,dE\Big\rvert
\le C_T\,\ell^{-1/2}\!\int_{\mathcal{I}}|\partial_E S|_2\,dE .
$$
