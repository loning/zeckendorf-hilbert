\begin{abstract}
Full-rank density matrix manifold $\mathcal{D}^{\rm full}_N=\{\rho>0,\ \mathrm{tr}\rho=1\}$ is open convex contractible, Uhlmann principal bundle admits global square-root section $w=\sqrt{\rho}$ on full domain, thus \textbf{absolute} integer-valued topological invariants are absent on full domain. This paper turns to \textbf{perforated relative topology}: In $N=5$ case, removing tubular neighborhood of three--two level gap closing set $\Sigma_{3|2}=\{\lambda_3=\lambda_4\}$ from $\mathcal{D}^{\rm full}_5$ yields perforated domain $\mathcal{D}^{\rm exc}$. On $\mathcal{D}^{\rm exc}$ construct rank $3/2$ subbundles $(\mathcal{E}_3,\mathcal{E}_2)$ via \textbf{Riesz spectral projection}, realizing principal bundle structure group reduction $U(5)\to U(3)\times U(2)$; further utilizing determinant balancing yields $S(U(3)\times U(2))$ reduction. We prove general group isomorphism

$$
S\!\big(U(m)\times U(n)\big)\ \cong\ \big(SU(m)\times SU(n)\times U(1)\big)\big/\mathbb{Z}_{\mathrm{lcm}(m,n)},
$$

with $(m,n)=(3,2)$ deriving $(SU(3)\times SU(2)\times U(1))/\mathbb{Z}_6$. Through \textbf{relative $K$-theory boundary map} unifying ``projection--Chern class'' with ``mass--clutching ($\det\widehat{\Phi}$ winding)'', on two-dimensional transverse $S^1$ obtain

$$
\mathrm{Ind}(\not{D}_A+\Phi)=\mathrm{wind}\,\det\widehat{\Phi}=\langle c_1(\mathcal{L}_\Phi),[S^1]\rangle .
$$

In $\mathbb{CP}^2$ spin$^c$/Dolbeault calibration compute $\mathrm{index}=3$ as ``three-generation prototype''. This paper provides \textbf{complete proofs} of all core propositions and theorems, with two \textbf{protocol-level} reproducible experimental/numerical schemes (purification interference loop and photonic Dirac--mass vortex). Appendices include: unified contour and global smoothness, group isomorphism gcd/lcm normalization and ``root selection'', rigorous proof of relative $K$-theory and Chern character commutative diagram, Fredholm construction for Callias/Anghel--Bunke index theorem, and one-page arithmetic derivation of minimal charge $1/6$ when $\Gamma=\mathbb{Z}_6$.

\textbf{Keywords}: Uhlmann principal bundle; perforated relative topology; Riesz projection; principal bundle reduction; $S(U(3)\times U(2))$; $\mathbb{Z}_6$ quotient; relative cohomology/$K$-theory; Dolbeault/spin$^c$ index; Callias/Anghel--Bunke index; determinant line bundle; line operator spectrum; reproducible experimental protocol
\end{abstract}

\section{Main Results (Statements)}

\begin{theorem}[A: Group Isomorphism, gcd/lcm Normalization]
Let $g=\gcd(m,n)$, $\ell=\mathrm{lcm}(m,n)=mn/g$. Homomorphism

$$
\varphi:\ SU(m)\times SU(n)\times U(1)\to S(U(m)\times U(n)),\quad
\varphi(A,B,z)=\mathrm{diag}\big(z^{n/g}A,\ z^{-m/g}B\big)
$$

is surjective with $\ker\varphi\simeq \mathbb{Z}_\ell$. Thus

$$
S\!\big(U(m)\times U(n)\big)\ \cong\ \big(SU(m)\times SU(n)\times U(1)\big)\big/\mathbb{Z}_\ell .
$$

Special case $(m,n)=(3,2)\Rightarrow \ell=6$.
\end{theorem}

\begin{proposition}[B: Partition Uniqueness]
Under constraint ``simple factors exactly $SU(3)$, $SU(2)$ retaining only one $U(1)$'', unique feasible partition of $U(5)$ is $5=3+2$.
\end{proposition}

\begin{theorem}[C: Relative Bridging]
Assume mass end term $\Phi$ invertible on $Y$, take unitization $\widehat{\Phi}:Y\to U(N)$. Then relative $K$-theory boundary image $\partial[\det\widehat{\Phi}]\in K^0(X,Y)$ equals projection line bundle $[\det\mathcal{E}_3]-[\det\mathcal{E}_2]$; on two-dimensional link

$$
\langle c_1(\mathcal{L}_\Phi),[S^2]\rangle=\langle c_1(\det\mathcal{E}_3),[S^2]\rangle\in\mathbb{Z} .
$$
\end{theorem}

\begin{theorem}[D: Callias/Anghel--Bunke]
If outer region invertibility $\Phi^2\ge cI$, $[\nabla,\Phi]\in L^\infty$, $\Phi\in W^{1,2}_{\rm loc}$ etc. hold, then

$$
\mathrm{Ind}(\not{D}_A+\Phi)=\deg\big(\widehat{\Phi}|_{S^{d-1}_\infty}\big)\in \pi_{d-1}(U).
$$

By Bott periodicity $\pi_k(U)=\mathbb{Z}$ ($k$ odd), $0$ ($k$ even), obtain: index possibly nonzero only when \textbf{transverse dimension $d$ is even}; when $d=2$

$$
\mathrm{Ind}=\frac{1}{2\pi i}\oint\mathrm{Tr}(\widehat{\Phi}^{-1}d\widehat{\Phi})
=\mathrm{wind}\,\det\widehat{\Phi}=\langle c_1(\mathcal{L}_\Phi),[S^1]\rangle .
$$
\end{theorem}

\begin{theorem}[E: $\mathbb{CP}^2$ Index]
$\mathrm{Td}(T\mathbb{CP}^2)=1+\tfrac{3}{2}H+H^2$, $\mathrm{ch}(\mathcal{O}(1))=1+H+\tfrac{1}{2}H^2$, thus
$\mathrm{index}\,\not{D}^{\mathcal{O}(1)}=3$.
\end{theorem}

\begin{corollary}[F: SM Global Group]
By Theorem A, Proposition B, obtain

$$
S(U(3)\times U(2))\ \cong\ \frac{SU(3)\times SU(2)\times U(1)}{\mathbb{Z}_6}.
$$

Appendix E further provides electric/magnetic charge lattice and arithmetic derivation of \textbf{minimal charge step $1/6$} for line operator spectrum when $\Gamma=\mathbb{Z}_6$.
\end{corollary}
