\begin{abstract}
Under four testable conditions---global unitarity/reversibility, finite memory/Markovizability, bandlimiting with Nyquist--Poisson--Euler--Maclaurin (NPE) finite-order closure, and time-ledger alignment---we construct functors $\mathfrak{F}:\mathbf{EBOC}\to\mathbf{GLS}$ and $\mathfrak{G}:\mathbf{GLS}\to\mathbf{EBOC}$ with natural transformations $\eta,\epsilon$, and prove that they are mutually equivalent in the observation category. The bridge is provided by the master-scale identity
$$
\boxed{\ \frac{\varphi'(E)}{\pi}=\frac{1}{2\pi}\mathrm{tr}\,\mathsf{Q}(E)\ },\qquad \boxed{\ \rho_{\text{rel}}(E)=\xi'(E)=-\frac{1}{2\pi}\mathrm{tr}\,\mathsf{Q}(E)\ },
$$
implying $\varphi'(E)/\pi=-\rho_{\text{rel}}(E)$ (where $\mathsf{Q}=-i\,S^\dagger \partial_E S$), and the windowed trace identity. The NPE discipline yields discretization and endpoint--tail closure in a non-asymptotic manner without introducing novelties. Thus, EBOC's ``static blocks--factor decoding--complexity/entropy ledger'' and GLS's ``unitary scattering--group delay--windowed measure'' are window-by-window consistent.
\end{abstract}

\noindent\textbf{Keywords:} EBOC--GLS equivalence; Observation category; Wigner--Smith time delay; Birman--Kre\u{\i}n spectral shift; Windowed trace identity; NPE closure; Factor entropy; Reversible cellular automata; Quantum cellular automata; Shannon--McMillan--Breiman theorem

\section{Notation, Objects, and Axioms}

\subsection{EBOC Objects and Observation Readouts}

An EBOC object is written as
$$
\mathcal{U}=(X_f,G,\rho,\Sigma,f,\pi,\mu,\nu),
$$
where $X_f\subset\Sigma^{\mathbb{Z}^{d+1}}$ is an SFT defined by a finite-neighborhood rule $f$; the time subaction is $\sigma_{\text{time}}$; $\pi$ is a finite-thickness foliation-factor decoding; $\mu$ is a $\sigma_{\text{time}}$-invariant ergodic measure; $\nu$ is a universal semimeasure. We adopt four testable axioms: A0 (static blocks and extendability), A1 (causal locality; finite propagation radius), A2 (observation $=$ factor decoding), A3 (information non-increase; factor entropy monotonicity). The Shannon--McMillan--Breiman (SMB) theorem and Brudno equality yield, on the time-slice family $W$ normalized by $L(W)=T$, alignment of complexity/entropy density almost everywhere:
$$
\lim_{|W|\to\infty}\frac{1}{L(W)}K\!\big(\pi(x)\mid W\big)= h_{\pi_\ast\mu}(\sigma_{\text{time}})\le h_{\mu}(\sigma_{\text{time}}).
$$
This extends to $\mathbb{Z}^d$ subshifts \cite{shields1996sandwich}.

\subsection{GLS Objects, Master Scale, and Windowed Readouts}

A GLS object is written as
$$
\mathfrak{U}=(\mathcal{H},S(E),\mathcal{W}),
$$
where $S(E)\in\mathsf{U}(N)$ is unitary and differentiable, $\mathsf{Q}(E)=-i\,S(E)^\dagger \tfrac{dS}{dE}(E)$ is the Wigner--Smith time-delay matrix, and $\mathrm{tr}\,\mathsf{Q}$ is the total group delay. The density--delay--phase relations satisfy
$$
\boxed{\ \frac{\varphi'(E)}{\pi}=\frac{1}{2\pi}\mathrm{tr}\,\mathsf{Q}(E)\ },\qquad \boxed{\ \rho_{\text{rel}}(E)=\xi'(E)=-\frac{1}{2\pi}\mathrm{tr}\,\mathsf{Q}(E)\ },
$$
and thus $\varphi'(E)/\pi=-\rho_{\text{rel}}(E)$. The connection to the spectral shift function $\xi$ via the Birman--Kre\u{\i}n formula $\det S(E)=e^{-2\pi i\,\xi(E)}$ yields $\xi'(E)=-\frac{1}{2\pi}\mathrm{tr}\,\mathsf{Q}(E)$ \cite{nollert2010birman}.

Given a window--kernel pair $(w_R,h)\in\mathcal{W}$, define the windowed readout
$$
\mathrm{Obs}(w_R,h;\rho)=\int_{\mathbb{R}} w_R(E)\,(h\star \rho_{\text{rel}})(E)\,dE,
$$
where $\star$ denotes the correlation operation: $(h\star \rho)(E):=(\check{h}\ast \rho)(E)=\int_{\mathbb{R}} h(t-E)\,\rho(t)\,dt$ with $\check{h}(E)=h(-E)$.

Let $P_\gamma$ be the orthogonal projection onto channel $\gamma$. Define the channel group-delay matrix
$$
\mathsf{Q}_\gamma(E):=P_\gamma\,\mathsf{Q}(E)\,P_\gamma,\qquad \mathrm{tr}\,\mathsf{Q}_\gamma(E)=\mathrm{Tr}\big(P_\gamma\,\mathsf{Q}(E)\big).
$$
Thus, the windowed group delay along channel $\gamma$ is
$$
T_\gamma[w_R,h]=\int_{\mathbb{R}} (w_R\ast \check{h})(E)\,\frac{1}{2\pi}\mathrm{tr}\,\mathsf{Q}_\gamma(E)\,dE.
$$

\subsection{NPE Finite-Order Error Theory and Sampling Closure}

If $\widehat{w}_R,\widehat{h}$ are bandlimited, choose the Nyquist step $\Delta$ to eliminate aliasing. Then for $f(E)=w_R(E)\,(h\star\rho_{\text{rel}})(E)$, we have exact sampling
$$
\int_{\mathbb{R}} f(E)\,dE=\Delta\sum_{n\in\mathbb{Z}} f(E_0+n\Delta).
$$
For near-bandlimited cases, aliasing terms are controlled by Poisson summation, and endpoint and tail terms are closed by finite-order Euler--Maclaurin (EM), with remainder having explicit Bernoulli--$\zeta$ bounds \cite{helfgott2013chapter}.

\section{Observation Category and Main Proposition}

\subsection{Observation Category}

Let the objects of $\mathbf{Obs}$ be ``system $+$ readout dictionary''; morphisms are factor/conjugacy mappings that preserve readouts (factor maps on the EBOC side; unitary conjugacy on the GLS side). Define
$$
\mathsf{Obs}_{\text{E}}(\mathcal{U}):=\big\{\mathcal{O}_{\pi,\varsigma},\text{and their limit statistics/entropy rates}\big\},
$$
$$
\mathsf{Obs}_{\text{G}}(\mathfrak{U}):=\big\{\mathrm{Obs}(w_R,h)\,,T_\gamma[w_R,h]\big\}.
$$

\subsection{Main Theorem (Observation Category Equivalence)}

\begin{theorem}[EBOC--GLS Observation Equivalence]
\label{thm:main_equivalence}
Under axioms A0--A3 and the ``four conditions,'' there exist functors $\mathfrak{F}:\mathbf{EBOC}\to\mathbf{GLS}$ and $\mathfrak{G}:\mathbf{GLS}\to\mathbf{EBOC}$ with natural transformations $\eta,\epsilon$ such that
$$
\mathsf{Obs}_{\text{G}}\!\big(\mathfrak{F}(\mathcal{U})\big)\cong \mathsf{Obs}_{\text{E}}(\mathcal{U}),\qquad \mathsf{Obs}_{\text{E}}\!\big(\mathfrak{G}(\mathfrak{U})\big)\cong \mathsf{Obs}_{\text{G}}(\mathfrak{U}),
$$
whereby $\mathfrak{F},\mathfrak{G}$ are mutually equivalent in $\mathbf{Obs}$.
\end{theorem}

\section{Technical Lemmas and Proofs}

\begin{lemma}[Master-Scale Unification Identity]
\label{lem:master_scale}
If $S(E)$ is unitary and differentiable, define $\mathsf{Q}=-i\,S^\dagger S'$. Then almost everywhere
$$
\rho_{\text{rel}}(E)=\xi'(E)=-\frac{1}{2\pi}\mathrm{tr}\,\mathsf{Q}(E),\quad \frac{\varphi'(E)}{\pi}=-\rho_{\text{rel}}(E),\quad \det S=e^{2i\varphi}.
$$
\end{lemma}

\begin{proof}
By the Birman--Kre\u{\i}n formula $\det S(E)=e^{-2\pi i\,\xi(E)}$ and $\partial_E\log\det S(E)=\mathrm{tr}\,(S^\dagger S')$, we obtain $\mathrm{tr}\,\mathsf{Q}(E)=-2\pi\,\xi'(E)$. Setting $\rho_{\text{rel}}=\xi'$ yields the first identity. If $\varphi=\tfrac{1}{2}\arg\det S$, then $\varphi'=\tfrac{1}{2}\mathrm{tr}\,\mathsf{Q}=-\pi\xi'$, thus $\varphi'/\pi=-\xi'=-\rho_{\text{rel}}$ yields the second identity \cite{nollert2010birman}.
\end{proof}

\begin{lemma}[Windowed Readout = Compressed Trace Identity]
\label{lem:trace_identity}
Let $(A,B)$ be self-adjoint operators with $A-B\in\mathcal{S}_1$. Set $f:=-(w_R\ast \check{h})$, and assume $f\in OL(\mathbb{R})$ with
$$
(w_R\ast \check{h})\in W^{1,1}(\mathbb{R}),\qquad (w_R\ast \check{h})(E)\,\xi(E)\xrightarrow[E\to\pm\infty]{}0.
$$
(For example: $w_R\in L^1\cap W^{1,1}$ with compact support or rapid decay, $h\in L^1$, then $(w_R\ast \check{h})\in W^{1,1}$; in typical scattering situations, $\xi$ has bounded variation and tends to a constant.) Define
$$
K_{w,h}:=f(A)-f(B)\qquad(\text{whose trace equals the Toeplitz/Berezin windowed compression defined by }(w_R,h)).
$$
Under these conditions, integration by parts has no boundary terms, thus
$$
\mathrm{Tr}\,K_{w,h}=\int_{\mathbb{R}} w_R(E)\,(h\star\rho_{\text{rel}})(E)\,dE.
$$
\end{lemma}

\begin{proof}
By the Lifshits--Kre\u{\i}n trace formula ($A-B\in\mathcal{S}_1$, $f\in OL$),
$$
\mathrm{Tr}\big(f(A)-f(B)\big)=\int_{\mathbb{R}} f'(E)\,\xi(E)\,dE.
$$
Following Section 0.2, define $(h\star\rho_{\text{rel}})=\check{h}\ast\rho_{\text{rel}}$ and $\rho_{\text{rel}}=\xi'$. Integration by parts yields
$$
\int_{\mathbb{R}} w_R(E)\,(h\star\rho_{\text{rel}})(E)\,dE=\int_{\mathbb{R}} (w_R\ast \check{h})(E)\,\xi'(E)\,dE
$$
$$
=-\int_{\mathbb{R}} (w_R\ast \check{h})'(E)\,\xi(E)\,dE=\int_{\mathbb{R}} f'(E)\,\xi(E)\,dE,
$$
establishing the conclusion. The unitary case can be handled via the Aleksandrov--Peller unit-circle version \cite{peller2017lifshits}.
\end{proof}

\begin{lemma}[Poisson--Nyquist and EM Closure]
\label{lem:poisson_nyquist}
If $\widehat{f}$ is supported on $(-\Omega,\Omega)$ and $\Delta\le \pi/\Omega$, then
$$
\int f=\Delta\sum_{n\in\mathbb{Z}} f(E_0+n\Delta).
$$
For near-bandlimited cases, adding Poisson aliasing bounds and EM bounded remainder yields non-asymptotic error budgets \cite{helfgott2013chapter}.
\end{lemma}

\begin{lemma}[SMB/Brudno and Factor Entropy Monotonicity]
\label{lem:smb_brudno}
For ergodic $(X_f,\mu)$ and factor $\pi$, almost everywhere
$$
\lim_{|W|\to\infty}\frac{1}{L(W)}K(\pi(x)\mid W)= h_{\pi_\ast\mu}(\sigma_{\text{time}})\le h_{\mu}(\sigma_{\text{time}}).
$$
\end{lemma}

\begin{proof}
The SMB theorem and its multi-dimensional extension yield entropy rates; the Brudno equality identifies complexity density; factor entropy monotonicity comes from sliding-block code theory (Lind--Marcus) \cite{shields1996sandwich}.
\end{proof}

\begin{lemma}[Markovizability and SFT Covering]
\label{lem:markovizability}
Observation sequences with finite memory (or finite memory after higher-order blocking) form sofic subshifts. Any sofic shift is a factor of some SFT, thus an SFT covering exists to realize the observation via local constraints \cite{lind1995introduction}.
\end{lemma}

\begin{lemma}[Reversibilization and QCA Structure Theorem]
\label{lem:qca_structure}
Any reversible classical CA (or a reversibilized system after Bennett register extension) can be quantized into a QCA unitary $U$ with strictly finite propagation speed and admits generalized Margolus partitioning. Its inverse is still a locally coupled QCA. Thus a Floquet unit step and scattering representation exist \cite{schumacher2004reversible}.
\end{lemma}

\section{Construction and Proof of Functor $\mathfrak{F}:\mathbf{EBOC}\to\mathbf{GLS}$}

\subsection{Reversibilization and Scatterization}

Apply a Bennett-style ``history register'' extension to $\mathcal{U}$ to obtain a reversible CA; quantize it following Schumacher--Werner into a finite-propagation QCA unitary $U$. Under appropriate local--external decomposition, construct a scattering matrix family $S(E)$ and Wigner--Smith matrix $\mathsf{Q}(E)$. By the QCA structure theorem, $U$ can be written as a two-step block unitary, guaranteeing finite propagation and structural reversibility \cite{schumacher2004reversible}.

\subsection{Foliation Decoding--Window Kernel Alignment}

Select foliation family $\varsigma$ and finite-thickness decoding $\pi$, matching kernel thickness with foliation step size, establishing the dictionary
$$
(\pi,\varsigma)\longleftrightarrow (w_R,h).
$$
By Lemmas \ref{lem:master_scale}--\ref{lem:trace_identity}, window-by-window readouts satisfy
$$
\mathcal{O}_{\pi,\varsigma}=\mathrm{Obs}(w_R,h;\rho)=\int_{\mathbb{R}} w_R(E)\,(h\star \rho_{\text{rel}})(E)\,dE,
$$
and all readouts are unified by the master scale $\varphi'(E)/\pi=-(\rho_{\text{rel}}(E))=\frac{1}{2\pi}\mathrm{tr}\,\mathsf{Q}(E)$ \cite{peller2017lifshits}.

\subsection{NPE Closure and ``No Novelties''}

If $\widehat{w}_R,\widehat{h}$ are bandlimited with Nyquist sampling, Poisson aliasing vanishes; for near-bandlimited cases, finite-order EM closes endpoints and tails. Lemma \ref{lem:poisson_nyquist} ensures the remainder has explicit bounds, the function class's analyticity/integrality does not deteriorate, satisfying ``singularity non-increase'' \cite{olver2008euler}.

\subsection{Time Ledger and No-Signaling}

Finite propagation $+$ unitarity yields no super-cone propagation (no signaling); windowed time $T_\gamma[w_R,h]$ is merely a calibration readout, with causal partial order determined by front time. By SMB/Brudno and factor entropy monotonicity, the EBOC time-slice ledger aligns with the GLS density--delay ledger \cite{cvitanovic2021chaos}.

\textbf{Definition:} $\mathfrak{F}(\mathcal{U}):=(\mathcal{H},S(E),\mathcal{W})$ with the agreement $\mathsf{Obs}_{\text{G}}(\mathfrak{F}(\mathcal{U}))=\mathsf{Obs}_{\text{E}}(\mathcal{U})$.

\section{Construction and Proof of Functor $\mathfrak{G}:\mathbf{GLS}\to\mathbf{EBOC}$}

\subsection{Bandlimited Discretization and Compressed Trace}

Let $f(E)=w_R(E)\,(h\star \rho_{\text{rel}})(E)$. For bandlimited + Nyquist,
$$
\mathrm{Obs}(w_R,h;\rho)=\int f(E)\,dE=\Delta\sum_{n\in\mathbb{Z}} f(E_0+n\Delta).
$$
For near-bandlimited, add finite-order EM remainder closure. At the operator level, windowed compression $K_{w,h}$ satisfies $\mathrm{Tr}\,K_{w,h}=\int f$ (Lemma \ref{lem:trace_identity}) \cite{helfgott2013chapter}.

\subsection{SFT-ization and Finite-Memory Decoding}

Quantize the sampled values $f(E_0+n\Delta)$ into a finite alphabet, supplemented with length-$M$ linear consistency constraints (convolution/prediction residual) to obtain a sofic subshift; take its SFT covering and obtain finite-thickness decoding $\pi$. By SMB/Brudno and factor entropy monotonicity, time-slice limit statistics align with $\mathrm{Obs}(w_R,h;\rho)$ \cite{lind1995introduction}.

\subsection{Causal and No-Signaling Discrete Realization}

Bandwidth $\Rightarrow$ finite correlation length $\Rightarrow$ finite propagation radius, which can be realized in SFT rules via local exclusivity and neighborhood constraints for causal partial order and no-signaling.

\textbf{Definition:} $\mathfrak{G}(\mathfrak{U}):=(X_f,G,\rho,\Sigma,f,\pi,\mu,\nu)$ with $\mathsf{Obs}_{\text{E}}(\mathfrak{G}(\mathfrak{U}))=\mathsf{Obs}_{\text{G}}(\mathfrak{U})$.

\section{Natural Transformations and Categorical Equivalence}

\begin{definition}[Natural Transformation $\eta:\mathrm{Id}_{\mathbf{EBOC}}\Rightarrow \mathfrak{G}\!\circ\!\mathfrak{F}$]
For each $\mathcal{U}$, let $\eta_\mathcal{U}$ ensure that for any aligned dictionary $(\pi,\varsigma)\leftrightarrow(w_R,h)$,
$$
\mathcal{O}_{\pi,\varsigma}=\mathrm{Obs}(w_R,h;\rho).
$$
This identity holds window-by-window within NPE budgets and commutes with factor morphisms.
\end{definition}

\begin{definition}[Natural Transformation $\epsilon:\mathfrak{F}\!\circ\!\mathfrak{G}\Rightarrow \mathrm{Id}_{\mathbf{GLS}}$]
For each $\mathfrak{U}$, adopt the gauge-phase-removed group-delay density identity:
$$
\boxed{\ -\frac{1}{2\pi}\mathrm{tr}\,\mathsf{Q}_{\mathfrak{F}(\mathfrak{G}(\mathfrak{U}))}(E)=\rho_{\text{rel}}(E)\ },
$$
thus all $\mathrm{Obs}(w_R,h)$, $T_\gamma[w_R,h]$ values are preserved.
\end{definition}

\begin{proposition}[Categorical Equivalence]
By the invariance of Lemmas \ref{lem:master_scale}--\ref{lem:poisson_nyquist} and the existence of Lemmas \ref{lem:smb_brudno}--\ref{lem:qca_structure}, $\eta,\epsilon$ are natural isomorphisms, thus $\mathfrak{F},\mathfrak{G}$ are mutually equivalent in $\mathbf{Obs}$ \cite{nollert2010birman}.
\end{proposition}

\section{Detailed Proofs of Key Steps}

\subsection{EBOC $\to$ GLS: Full Details of Kernel Consistency}

Take any bandlimited $(w_R,h)$. By Lemma \ref{lem:master_scale}, the master-scale identity gives $\rho_{\text{rel}}=-\frac{1}{2\pi}\mathrm{tr}\,\mathsf{Q}$ and $\frac{\varphi'}{\pi}=-\rho_{\text{rel}}$. By Lemma \ref{lem:trace_identity},
$$
\mathrm{Obs}(w_R,h;\rho)=\mathrm{Tr}\,K_{w,h}.
$$
On the EBOC side, $\pi$ is a finite-thickness decoding, $\mu$ is ergodic, and $L(W)=T$ is the time length. By Birkhoff/SMB and Brudno, $\mu$-almost everywhere (assuming $\int_{\mathbb{R}} w_R(E)\,dE\neq 0$),
$$
\lim_{|W|\to\infty}\frac{1}{L(W)}\sum_{t\in W}\mathcal{O}_{\pi,\varsigma}\!\big(\sigma_{\text{time}}^t x\big)=\frac{1}{\int_{\mathbb{R}} w_R(E)\,dE}\mathrm{Tr}\,K_{w,h}
$$
$$
=\frac{1}{\int_{\mathbb{R}} w_R(E)\,dE}\int_{\mathbb{R}} w_R(E)\,(h\star\rho_{\text{rel}})(E)\,dE.
$$
Thus $\mathsf{Obs}_{\text{G}}(\mathfrak{F}(\mathcal{U}))\cong \mathsf{Obs}_{\text{E}}(\mathcal{U})$ window-by-window consistent on the time-density scale \cite{peller2017lifshits}.

\subsection{GLS $\to$ EBOC: Construction Details of SFT-ization}

Let $f(E)=w_R(E)\,(h\star \rho_{\text{rel}})(E)$. Bandlimited + Nyquist:
$$
\int f=\Delta\sum_{n\in\mathbb{Z}} f(E_0+n\Delta).
$$
For near-bandlimited,
$$
\int_{\mathbb{R}} f(E)\,dE= \Delta\sum_{|n|\le N} f(E_0+n\Delta)- 2\pi\sum_{k\ne 0}\widehat{f}\!\left(\frac{2\pi k}{\Delta}\right)e^{i\,2\pi k E_0/\Delta}+ R_{\text{EM}}^{(m)},
$$
where $R_{\text{EM}}^{(m)}$ is the EM endpoint and tail remainder for closing $\Delta\sum_{n\in\mathbb{Z}}f(E_0+n\Delta)$ to $|n|\le N$; if $\widehat{f}$ is supported on $(-\pi/\Delta,\pi/\Delta)$, the aliasing term vanishes, degenerating to the exact sampling identity. Quantize the finite-precision sampled values $f(E_0+n\Delta)$ into an alphabet; under linear filter consistency (length $M$), define a sofic subshift; take an SFT covering to obtain $X_f$ and finite-thickness decoding $\pi$. By Lemma \ref{lem:trace_identity},
$$
\mathrm{Obs}(w_R,h;\rho)=\mathrm{Tr}\,K_{w,h}=\lim_{k\to\infty}\mathcal{O}_{\pi,\varsigma}\!\big(x\!\mid_{W_k}\big),
$$
yielding $\mathsf{Obs}_{\text{E}}(\mathfrak{G}(\mathfrak{U}))\cong \mathsf{Obs}_{\text{G}}(\mathfrak{U})$ \cite{helfgott2013chapter}.

\subsection{Preservation of Time Ledger and No-Signaling}

The finite propagation and unitarity of QCA ensure no super-cone propagation; windowing order affects only calibration, not master-scale density. In the EBOC$\leftrightarrow$GLS round-trip, causal fronts and partial order are preserved, while $T_\gamma$ serves only as operational calibration \cite{schumacher2004reversible}.

\section{Necessity--Sufficiency Argument}

\textbf{Sufficiency:}
\begin{enumerate}
\item[(a)] Reversibilization and quantization generate $S(E)$, $\mathsf{Q}(E)$ (Lemma \ref{lem:qca_structure});
\item[(b)] Window--kernel readouts and decoding readouts are linearly consistent on the master scale (Lemmas \ref{lem:master_scale}--\ref{lem:trace_identity});
\item[(c)] Poisson + EM NPE closure yields non-asymptotic error budgets without introducing novelties (Lemma \ref{lem:poisson_nyquist});
\item[(d)] SMB/Brudno and factor entropy monotonicity align complexity/entropy ledgers with density--delay ledgers (Lemma \ref{lem:smb_brudno}).
\end{enumerate}

\textbf{Necessity:}
Without global unitarity/reversibility, $S(E),\mathsf{Q}(E)$ cannot be well-defined; without finite memory/Markovizability, finite-thickness decoding cannot align window--kernel dictionaries; without bandlimiting or violating Nyquist, Poisson aliasing destroys window-by-window consistency; without $L(W)=T$ ledger alignment, SMB/Brudno cannot match windowed traces.

\section{Error Budget (NPE Discipline)}

For any $(w_R,h)$ and near-bandlimited $\rho_{\text{rel}}$, let $f=w_R\cdot(h\star \rho_{\text{rel}})$, take sampling step $\Delta$, truncation $N$, EM order $m$, and set
$$
[a,b]=\Big[E_0-\Big(N+\tfrac{1}{2}\Big)\Delta,\ \ E_0+\Big(N+\tfrac{1}{2}\Big)\Delta\Big],\quad f\in C^{2m}([a,b]),\ \ \widehat{f}\in L^1(\mathbb{R}).
$$
Then we have the following aliasing + EM bound:
$$
\Bigl|\int f-\Delta\!\sum_{|n|\le N}f(E_0+n\Delta)\Bigr|\le 2\pi\sum_{k\ne 0}\Bigl|\widehat{f}\!\Bigl(\frac{2\pi k}{\Delta}\Bigr)\Bigr|
$$
$$
+\frac{|B_{2m}|}{(2m)!}\bigl|f^{(2m-1)}(b)-f^{(2m-1)}(a)\bigr|+\frac{2\zeta(2m)}{(2\pi)^{2m}}\!\int_a^b\!|f^{(2m)}(E)|\,dE.
$$
Choose $\Delta,N,m$ to reach the target precision threshold \cite{olver2008euler}.

\section{Conclusive Theorem (Formal Statement)}

\begin{theorem}[EBOC--GLS Categorical Equivalence]
\label{thm:final}
Under axioms A0--A3 and the ``four conditions,'' $\mathfrak{F},\mathfrak{G}$ and natural transformations $\eta,\epsilon$ ensure
$$
\boxed{\ \mathfrak{F}:\mathbf{EBOC}\rightleftarrows \mathbf{GLS}:\mathfrak{G}\ \text{are mutually equivalent in}\ \mathbf{Obs}\ }.
$$
The core identities are
$$
\boxed{\ \frac{\varphi'}{\pi}=\frac{1}{2\pi}\mathrm{tr}\,\mathsf{Q}\ },\quad \boxed{\ \rho_{\text{rel}}=\xi'=-\frac{1}{2\pi}\mathrm{tr}\,\mathsf{Q}\ },
$$
thus $\frac{\varphi'}{\pi}=-\rho_{\text{rel}}$, and
$$
\boxed{\ \mathrm{Tr}\,K_{w,h}=\int w_R(E)\,(h\star\rho_{\text{rel}})(E)\,dE\ }.
$$
The first identity follows from Birman--Kre\u{\i}n and Wigner--Smith structure; the second from the Lifshits--Kre\u{\i}n trace formula and windowed compression. The NPE discipline provides non-asymptotic closure and avoids novelties.
\end{theorem}

\section{Discussion and Conclusions}

This paper establishes a categorical equivalence between EBOC (Eternal Block of Contents) and GLS (Generalized Light Structures) within the observation category framework. Our main contributions include:

\begin{enumerate}
\item The master-scale unification identity linking Wigner--Smith time delay, Birman--Kre\u{\i}n spectral shift, and scattering phase derivative.

\item The windowed trace identity establishing the precise correspondence between factor decodings and windowed observables.

\item The construction of explicit functors $\mathfrak{F}$ and $\mathfrak{G}$ bridging reversible cellular automata and unitary scattering systems.

\item The NPE (Nyquist--Poisson--Euler--Maclaurin) discipline providing finite-order, non-asymptotic error control without introducing singularities.

\item The verification that complexity/entropy ledgers on the EBOC side and density--delay measures on the GLS side are window-by-window consistent through SMB/Brudno alignment.
\end{enumerate}

The equivalence demonstrates that static-block structures with factor decoding and dynamic scattering structures with windowed measurements are dual descriptions of the same underlying observational content. This provides a rigorous mathematical foundation for treating reversible computation and quantum scattering as observation-equivalent frameworks.

Future directions include extending the framework to non-reversible dissipative systems, incorporating continuous-time limits, and exploring applications to quantum information theory and black hole physics.
