\begin{abstract}
We establish a rigorous theory unifying ``finite throughput--queueing delay--apparent dilation (redshift)'' within windowed scattering and information geometry (WSIG) scale system. In pure theoretical language of operator--measure--function, using ``trinity'' scale identity $\varphi'(E)/\pi=\rho_{\rm rel}(E)=-\frac{1}{2\pi}\operatorname{tr}\mathsf Q(E)$ as mother scale, define observer window's Toeplitz/Berezin compression $K_{w,h}$ and its readout functional, characterize ``finite throughput'' abstract form via passivity and causality-preserving scattering axioms. Prove three main theorems: (T1) \textbf{Throughput--Delay Dilation Theorem}: Under passive load monotonicity, any windowed group delay readout monotonically non-decreasing with load; when baseline readout $\Phi_w[\lambda_0]>0$, induces Mellin dilation of energy and time scales scale-equivalent to redshift factor $1+z_w=\Delta_w\ge1$; (T2) \textbf{Group Delay--Bandwidth Product Upper Bound}: Windowed energy spectrum total readout upper-bounded by product of ``group delay flux'' and ``effective bandwidth constant'', with non-asymptotic error bound from Nyquist--Poisson--Euler--Maclaurin (NPE) three-term closure; (T3) \textbf{Scale Gauge Equivalence}: Cosmological expansion and internal resolution enhancement scale-equivalent to gauge pairing $a(t)=R(t)^{-1}$, all readouts remaining invariant on same mother scale. Further provide ``front slope--group delay'' correspondence at reversible cellular automaton (RCA) dynamics layer, embedding queueing--throughput--delay discrete dynamics into EBOC static block encoding. Throughout use only finite-order Euler--Maclaurin and Poisson discipline, maintaining ``singularity non-increasing/poles = principal scales'' error control; any numerical and experimental interfacing relegated to appendices.
\end{abstract}

\noindent\textbf{Keywords:} Finite throughput; Group delay dilation; WSIG; EBOC; RCA; NPE error closure; Redshift equivalence; Scale gauge

\section*{Notation \& Axioms / Conventions}

\begin{enumerate}
\item \textbf{Scale Identity (Trinity):} For absolutely continuous spectrum a.e.
\begin{equation}
\boxed{\ \frac{\varphi'(E)}{\pi}=\rho_{\rm rel}(E)=-\frac{1}{2\pi}\operatorname{tr}\mathsf Q(E)\ },\qquad
\mathsf Q(E)=-\,i\,S(E)^\dagger S'(E),\ S(E)\in U(N),
\end{equation}
where $\varphi$ is total scattering phase, $\rho_{\rm rel}$ relative state density (Birman--Kreĭn perspective), $\mathsf Q$ Wigner--Smith group delay matrix. By Birman--Kreĭn formula $\det S(E)=\exp(-2\pi i\,\xi(E))$ and $-i\,\partial_E\log\det S(E)=\operatorname{tr}\mathsf Q(E)$ obtain $\operatorname{tr}\mathsf Q(E)=-2\pi\,\xi'(E)$, thus above holds ($\xi$ spectral shift function).

\item \textbf{Observer Window and Compression:} Take non-negative weight window $w\in L^1(\mathbb R)\cap L^\infty$ and smoother $h$ (approximating identity, Carleson regular). Define Toeplitz/Berezin compression $K_{w,h}$ acting on energy spectrum symbol $F(E)$ windowed readout
\begin{equation}
\langle F\rangle_{w}:=\int_{\mathbb R} w(E)\,\operatorname{tr}F(E)\,dE,\qquad
\langle F\rangle_{w,h}:=\operatorname{tr}(K_{w,h}F),
\end{equation}
differing under NPE discipline only by finite-order Poisson+Euler--Maclaurin error. Characterization of Toeplitz/Berezin and Carleson embedding see recent developments in weighted Bergman/Hardy systems.

\item \textbf{Passivity and Monotonicity:} Load represented by unitarity-preserving scattering family $S_\lambda(E)\in U(N)$ ($\lambda\in[0,1]$). Passivity axiom written as
\begin{equation}
\partial_\lambda \mathsf Q_\lambda(E)\succeq 0\ \text{ a.e.}
\end{equation}
Throughout derivations use only this Loewner monotonicity, no longer claim sign equivalence with $\partial_\lambda\rho_{\rm rel}$. Supportable by monotonicity of spectral shift function $\xi(E;\lambda)$ for monotonically coupled self-adjoint spectral pairs $(H_\lambda=H_0+\lambda V,\ V\ge0)$ with $\lambda$.

\item \textbf{NPE Three-Term Closure (Finite-Order):} Discrete--continuous error decomposition
\begin{equation}
\mathcal E_{\rm total}=\mathcal E_{\rm alias}+\mathcal E_{\rm BL}+\mathcal E_{\rm tail},
\end{equation}
where $\mathcal E_{\rm alias}$ from Poisson spectral aliasing, $\mathcal E_{\rm BL}$ finite-order Euler--Maclaurin boundary layer term, $\mathcal E_{\rm tail}$ controlled by window decay and target function step. Standard references for Poisson and Euler--Maclaurin (including Bernoulli constants) see cited sources.

\item \textbf{Scale Gauge:} Dual gauge of external scale factor $a(t)$ and internal resolution constant $R(t)$
\begin{equation}
a(t)=R(t)^{-1},\qquad \kappa(t):=\dot a/a=-\dot R/R,
\end{equation}
implements Mellin dilation on energy mother scale, redshift satisfies $1+z=a(t_0)/a(t_e)=R(t_e)/R(t_0)$.
\end{enumerate}

\section{Windowed Readout, Group Delay Flux, and Effective Bandwidth}

Define windowed group delay flux
\begin{equation}
\Phi_w:=\frac{1}{2\pi}\int_{\mathbb R} w(E)\,\operatorname{tr}\mathsf Q(E)\,dE
=-\langle \rho_{\rm rel}\rangle_w.
\end{equation}
Define window's effective bandwidth constant
\begin{equation}
B_w:=|w|_{L^1}\cdot|\widehat{w}|_{\rm pack},
\end{equation}
where $\widehat{w}$ corresponding Fourier--Mellin transform, $|\cdot|_{\rm pack}$ controlled by Nyquist/Landau density (for bandlimited or tight frame window families can be given by upper/lower bound constants). If $0\preceq F(E)\preceq \alpha(E)\mathbf 1$, then
\begin{equation}
\langle F\rangle_w\le N\,|\alpha|_{L^\infty(\operatorname{supp}w)}\cdot |w|_{L^1}.
\end{equation}
General case $\mathsf Q$ not necessarily positive, thus cannot take $F=(2\pi)^{-1}\mathsf Q$; can use
\begin{equation}
\boxed{\ \Big|\Big\langle \tfrac{1}{2\pi}\mathsf Q\Big\rangle_w\Big|
\le \frac{N}{2\pi}\,\big|\mathsf Q\big|_{L^\infty(\operatorname{supp}w)}\,|w|_{L^1}\ }
\end{equation}
as crude upper bound. Necessary conditions for sampling/frame density and bandwidth packing see Landau and subsequent generalizations.

\section{Throughput Abstraction: Operator--Measure--Function Paradigm}

Let observation triple $(\mathcal H,w,S)$. Write scale measure $d\mu_{\mathsf Q}:=(2\pi)^{-1}\operatorname{tr}\mathsf Q(E)\,dE$. Define energy readout
\begin{equation}
\mathscr R_w:=\int w(E)\,\rho_{\rm rel}(E)\,dE
=-\frac{1}{2\pi}\int w(E)\,\operatorname{tr}\mathsf Q(E)\,dE
=-\int w(E)\,d\mu_{\mathsf Q}(E)=-\Phi_w.
\end{equation}
``Finite throughput'' abstracted as Carleson-type constraint on $\mu_{\mathsf Q}$: exists constant $C_{\rm th}$ and energy domain partition $\{\Omega_k\}$ such that
\begin{equation}
|\mu_{\mathsf Q}|(\Omega_k)\le C_{\rm th}\,\Lambda(\Omega_k),
\end{equation}
where $\Lambda$ reference measure induced by window family (given by Nyquist density or frame density). Corresponding Toeplitz/Berezin type embedding in weighted function spaces characterizable by Carleson condition.

\section{Main Theorem I: Throughput--Delay Dilation Theorem}

\begin{theorem}[Load Monotonicity and Dilation Factor]
Let $S_\lambda(E)$ satisfy passivity axiom $\partial_\lambda \mathsf Q_\lambda(E)\succeq 0$ a.e., and window $w\ge 0$ compactly supported or satisfying frame regularity and NPE conditions. Then for any $0\le \lambda_0<\lambda_1\le 1$,
\begin{equation}
\Phi_w[\lambda_1]-\Phi_w[\lambda_0]
=\frac{1}{2\pi}\!\int w(E)\,\operatorname{tr}\big(\mathsf Q_{\lambda_1}(E)-\mathsf Q_{\lambda_0}(E)\big)\,dE\ \ge 0.
\end{equation}
Define scale's ``delay dilation factor'' (\textbf{when $\Phi_w[\lambda_0]>0$})
\begin{equation}
\Delta_w(\lambda_1,\lambda_0):=\frac{\Phi_w[\lambda_1]}{\Phi_w[\lambda_0]}\ \ge\ 1.
\end{equation}
General case always-valid statement is
\begin{equation}
\Phi_w[\lambda_1]-\Phi_w[\lambda_0]\ \ge\ 0,
\end{equation}
equivalent to $\mathscr R_w$'s non-increasing (because $\mathscr R_w=-\Phi_w$). By Trinity, $\Delta_w$ simultaneously characterizes phase density derivative and relative state density elongation, inducing energy scale--time scale Mellin dilation, equivalent to redshift factor $1+z_w=\Delta_w(\lambda_1,\lambda_0)\ge1$. Root lies in spectral shift function $\xi(E;\lambda)$ monotonic with non-negative coupling, and $\operatorname{tr}\mathsf Q(E)=-2\pi\,\partial_E\xi(E)$.
\end{theorem}
\begin{proof}
Loewner order monotonicity at each $E$ and $w\ge0$ integration immediately yields non-decreasing. By Birman--Kreĭn and Trinity linear relations, monotonicity equivalently transmitted between $\varphi'$ and $\operatorname{tr}\mathsf Q$; on $\rho_{\rm rel}$ differs only by fixed negative sign ($\rho_{\rm rel}=-\tfrac{1}{2\pi}\operatorname{tr}\mathsf Q$), thus above non-decreasing unified expressed as $\Phi_w$.
\end{proof}

\begin{corollary}[Window--Band Domain Locality]
If $w$ supported on bandlimited region $\Omega$, then $z_w$ is local redshift for that domain; for mutually non-overlapping $\{\Omega_j\}$ with decomposition windows $\{w_j\}$ forming tight frame,
\begin{equation}
1+z_{\rm global}=\sum_j \omega_j\,(1+z_{w_j}),\qquad
\omega_j=\frac{\Phi_{w_j}[\lambda_0]}{\sum_k \Phi_{w_k}[\lambda_0]},
\end{equation}
giving global dilation as weighted combination of local dilations; if baseline state $\mu_{\mathsf Q}[\lambda_0]$ is non-negative measure (equivalent to $\Phi_{w_j}[\lambda_0]\ge0$ for all $j$), this combination is convex combination.
\end{corollary}

\section{Main Theorem II: Group Delay--Bandwidth Product Upper Bound}

\begin{theorem}[Finite Window Energy Spectrum Conservation Upper Bound]
Under §2's Carleson-type throughput constraint, window family tight frame, and NPE validity, for any $S(E)$ and window $w$
\begin{equation}
\boxed{\ \big|\Phi_w\big|\ \le\ C_{\rm frame}\,C_{\rm th}\,B_w\,\big(1+\big|\varepsilon_{\rm NPE}\big|\big)\ },
\end{equation}
where $C_{\rm frame}$ given by window family upper/lower bound constants, $B_w$ effective bandwidth constant, $\varepsilon_{\rm NPE}$ finite-order error term satisfying
\begin{equation}
|\varepsilon_{\rm NPE}|\ \le\ C_1\,\mathcal E_{\rm alias}+C_2\,\mathcal E_{\rm BL}^{(m)}+C_3\,\mathcal E_{\rm tail}^{(\beta)}.
\end{equation}
\end{theorem}
\begin{proof}[Proof Sketch]
Along Berezin compression commutator inequalities and Carleson embedding control $\mu_{\mathsf Q}$, window family frame upper/lower bounds connect local energy spectrum with global readout, finally use finite-order Poisson dealiasing and Euler--Maclaurin boundary layer absorb discrete--continuous difference; singularity non-increasing ensures principal scale poles not amplified.
\end{proof}

\begin{corollary}[Optimal Window's Variational Tendency]
Under fixed $C_{\rm th}$ and resource constraints $|w|_{L^1}\,,|\widehat w|_{\rm pack}$, $w^\star$ maximizing $\Phi_w$ tends toward tight frame and near-minimum uncertainty window; if bandlimited domain invariant, $w^\star$ nearly constant amplitude within domain, minimum boundary layer cost at boundaries. Its frame/density conditions compatible with Wexler--Raz biorthogonality and Balian--Low obstructions.
\end{corollary}

\section{Main Theorem III: Scale Gauge Equivalence and Redshift = Resolution Enhancement}

\begin{theorem}[Gauge Equivalence]
Let $a(t),R(t)$ satisfy $a(t)=R(t)^{-1}$ and $\kappa(t)=\dot a/a=-\dot R/R$. On Trinity mother scale, for any window $w$ and energy readout
\begin{equation}
\mathscr R_w(t)=\int w(E)\,\rho_{\rm rel}(E;t)\,dE=-\frac{1}{2\pi}\int w(E)\,\operatorname{tr}\mathsf Q(E;t)\,dE,
\end{equation}

\begin{equation}
\mathscr R_w(t_0)=\mathscr R_{(1+z)^{-1}\,w\circ \mathcal D_{1+z}}(t_e),\qquad \mathcal D_{1+z}:E\mapsto \frac{E}{1+z},\quad 1+z=\frac{a(t_0)}{a(t_e)}=\frac{R(t_e)}{R(t_0)}.
\end{equation}
Thus ``expansion'' and ``resolution enhancement'' merely gauge restatements under same mother scale, not changing any windowed readout's ontological meaning.
\end{theorem}
