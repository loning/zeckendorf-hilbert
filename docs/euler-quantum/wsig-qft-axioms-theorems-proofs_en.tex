\begin{abstract}
This paper constructs and rigorizes \textbf{WSIG-QFT (Windowed Scattering \& Information-Geometry Quantum Field Theory)}: under weighted Mellin--logarithmic model and de Branges--Kreĭn (DBK) canonical system, uses \textbf{Weyl--Heisenberg} kinematic scale for ``phase--scale'', connects scattering phase derivative with (relative) spectral density via \textbf{Birman--Kreĭn (BK)--Wigner--Smith (WS)} chain, realizes \textbf{Born probability = relative entropy minimization} through \textbf{Csiszár-type I-projection}, implements \textbf{pointer basis = spectral minimum of readout quadratic form} via \textbf{Ky-Fan spectral minimum}, provides non-asymptotic error closure and bandlimited-sampling criterion through \textbf{Nyquist--Poisson--Euler--Maclaurin (NPE)}. For multi-channel establishes \textbf{windowed BK identity} and \textbf{multi-window frame--Wexler--Raz} synergy conditions, giving verifiable premises, explicit statements and complete proofs (based on recognized criteria).

\textbf{Keywords}: Windowed readout; de Branges--Kreĭn canonical system; Weyl--Heisenberg representation; Birman--Kreĭn; Wigner--Smith delay; I-projection; Wexler--Raz; Nyquist--Poisson--Euler--Maclaurin (NPE) error closure

\textbf{MSC}: 81Txx; 47Bxx; 46E22; 42C15
\end{abstract}

\section{Setup and Notation}

\subsection{Logarithmic--Mellin Model and Mirror Involution}

Take $\mathcal{H}_a=L^2(\mathbb{R}_+,x^{a-1}dx)$, let $x=e^t$ then isometric with $L^2(\mathbb{R})$. Define modulation/scale action

$$
(U_\tau f)(x)=x^{i\tau}f(x),\qquad (V_\sigma f)(x)=e^{\sigma a/2}f(e^\sigma x),
$$

satisfying $V_\sigma U_\tau=e^{i\tau\sigma}U_\tau V_\sigma$ (Weyl relation). Mirror involution $(Jf)(x)=x^{-a}f(1/x)$ unitary, Mellin transform satisfies $\mathcal{M}_a[Jf](s)=\mathcal{M}_a[f](a-s)$. This symmetry given by standard Mellin identities appearing in handbooks and DLMF entries.

\subsection{DBK Canonical System and Herglotz--Weyl Dictionary}

Take half-axis canonical system $JY'(t,z)=zH(t)Y(t,z)$ ($H\succeq0$ integrable, $J=\tiny\begin{pmatrix}0&-1\\1&0\end{pmatrix}$). Its Weyl--Titchmarsh function $m(z)$ is Herglotz, non-tangential boundary imaginary part gives spectral density $\rho(E)=\pi^{-1}\Im m(E+i0)$; every Herglotz function originates from some trace-normed canonical system (de Branges theorem).

\subsection{Scattering Data and Phase--Delay Matrix}

Set scatterable pair $(H_0,H)$ satisfying trace-class perturbation premise; S-matrix $S(E)$'s Wigner--Smith delay matrix $Q(E)=-iS(E)^\ast \partial_E S(E)$ well-defined, eigenvalues are ``intrinsic delay times''.

\section{WSIG-QFT Axioms}

\begin{axiom}[Weyl--Heisenberg Covariance and Mirror]
\label{ax:weyl-mirror}
Physical observable phase--scale action realized by projective unitary representation of $(U_\tau,V_\sigma)$, mirror $J$ realizes $s\mapsto a-s$ completion symmetry (Mellin side).
\end{axiom}

\begin{axiom}[Windowed Readout]
\label{ax:windowed-readout}
Any real readout equivalent to energy-side convolution--weighted linear functional

$$
\mathcal{R}[F;\rho_\star]\equiv \int_{\mathbb{R}} F(E)\,\rho_\star(E)\,dE,\qquad F:=h\!*\!w_R,
$$

where $h$ frontend kernel, $w_R$ even window, $\rho_\star=\rho$ or relative density $\rho-\rho_0$.
\end{axiom}

\begin{axiom}[Phase--Density Scale]
\label{ax:phase-density}
Under BK and WS chain, almost everywhere

$$
\frac{1}{2\pi}\operatorname{tr}Q(E)=\xi'(E)=\operatorname{tr}\big(\rho-\rho_0\big)(E),\qquad \boxed{\det S(E)=e^{2\pi i\,\xi(E)}},
$$

where positive sign convention completely consistent with $\xi'=\tfrac{1}{2\pi}\operatorname{tr}Q$ and single-channel $\varphi'(E)=\pi\,\rho_{\mathrm{rel}}(E)$ ($S=e^{2i\varphi}$).
\end{axiom}

\begin{axiom}[Probability--Information Consistency]
\label{ax:prob-info}
Solution minimizing KL-divergence over linear moment constraint family equivalent to Born probability; necessary and sufficient conditions given by Csiszár's I-projection geometry and Pythagorean identity.
\end{axiom}

\begin{axiom}[NPE Non-Asymptotic Closure]
\label{ax:npe-closure}
For uniform sampling/numerical quadrature of $F=h\!*\!w_R$, error decomposes as \textbf{alias (Poisson) + EM Bernoulli layer + tail} three terms; if $\operatorname{supp}\widehat{F}\subset[-\Omega_F,\Omega_F]$ and $\Delta\le \pi/\Omega_F$, alias term is $0$.
\end{axiom}

\section{Kinematics and Mirror Kernel}

\begin{theorem}[CCR--Weyl Relation and Logarithmic Representation Equivalence]
\label{thm:ccr-weyl}
Let $U_\tau=e^{i\tau A}$, $V_\sigma=e^{i\sigma B}$, where on core $\mathcal{D}:=C_c^\infty(\mathbb{R}_+)$

$$(Af)(x)=(\log x)f(x),\qquad (Bf)(x)=-i\big(x\partial_x+\tfrac{a}{2}\big)f(x).$$

Then on common dense core $\mathcal{D}$ have $[A,B]=iI$, after closure $[\overline{A},\overline{B}]=iI$, exponential forms give $V_\sigma U_\tau=e^{i\tau\sigma}U_\tau V_\sigma$. Via $x=e^t$ isometry, unitarily equivalent to modulation--translation representation of $L^2(\mathbb{R})$.
\end{theorem}

\begin{proof}
Stone theorem gives strongly continuous one-parameter groups and generators; direct calculation yields Weyl relation; isometric map given by $L^2(\mathbb{R}_+,x^{a-1}dx)\simeq L^2(\mathbb{R})$ and Mellin--Fourier interconversion. \end{proof}

\begin{theorem}[Mirror Kernel and Completed Function]
\label{thm:mirror-kernel}
If $K(x)=x^{-a}K(1/x)$ and $K\in L^1(\mathbb{R}_+,x^{a-1}dx)$, then Mellin transform $\Phi(s)=\int_0^\infty K(x)x^{s-1}dx$ satisfies $\Phi(s)=\Phi(a-s)$. Multiplying by symmetry factor $r(s)$ gives completed function $\Xi(s)=r(s)\Phi(s)$.
\end{theorem}

\begin{proof}
Direct from definition of $J$ and $\mathcal{M}_a[Jf](s)=\mathcal{M}_a[f](a-s)$. \end{proof}

\section{Dynamics: Phase--Density--Delay}

\begin{theorem}[Phase Derivative = (Relative) Spectral Density]
\label{thm:phase-derivative-density}
Set $(H_0,H)$ self-adjoint pair with $H-H_0\in \mathfrak{S}_1$. Denote spectral shift function $\xi$ and S-matrix $S(E)$. Then a.e. $E$ have

$$
\boxed{\det S(E)=e^{2\pi i\,\xi(E)}},\qquad \xi'(E)=\frac{1}{2\pi}\operatorname{tr}Q(E)=\operatorname{tr}(\rho-\rho_0)(E),
$$

single-channel $S(E)=e^{2i\varphi(E)}$ and $\varphi'(E)=\pi\,\rho_{\mathrm{rel}}(E)$.
\end{theorem}

\begin{proof}
First formula is Birman--Kreĭn formula; second from $Q(E)=-iS^*\partial_ES$ and $\partial_E\arg\det S(E)=\operatorname{tr}Q(E)$; equivalence of $\xi'$ with relative local density of states (relative LDOS) see spectral shift--trace formula (next section). Single-channel case substitute $S=e^{2i\varphi}$ proves. \end{proof}

\begin{proposition}[Threshold and Phase Critical Alignment]
\label{prop:threshold-alignment}
If at threshold $E_0$ have $\rho_{\mathrm{rel}}(E_0)=0$, then $\varphi'(E_0)=0$.
\end{proposition}

\begin{proof}
Direct from Theorem~\ref{thm:phase-derivative-density} single-channel formula $\varphi'=\pi\,\rho_{\mathrm{rel}}$. \end{proof}

\section{Windowed Trace Formula and Windowed BK Identity}

\begin{theorem}[Lifshits--Kreĭn Trace Formula--Windowed Version]
\label{thm:lifshits-krein-windowed}
Set $f\in \OL(\mathbb{R})$ (operator Lipschitz), take primitive of $f=(h\!*\!w_R)$ such that $f' =F$. Then

$$
\operatorname{tr}\!\big(f(H)-f(H_0)\big)=\int_{\mathbb{R}} f'(E)\,\xi(E)\,dE
= \int_{\mathbb{R}} F(E)\,\xi(E)\,dE.
$$
\end{theorem}

\begin{proof}
For paired self-adjoint operators with $H-H_0\in\mathfrak{S}_1$, Lifshits--Kreĭn trace formula holds on $\OL$ class; setting $f' =F$ yields ``windowed trace''. \end{proof}

\begin{theorem}[Windowed Birman--Kreĭn Identity]
\label{thm:windowed-bk}
Under Theorem~\ref{thm:lifshits-krein-windowed} premises, integration by parts using $\boxed{\det S(E)=e^{2\pi i\,\xi(E)}}$ gives

$$
\int_{\mathbb{R}} F(E)\,\xi'(E)\,dE
= -\frac{1}{2\pi i}\int_{\mathbb{R}} F'(E)\,\log\det S(E)\,dE
= -\frac{1}{2\pi i}\int_{\mathbb{R}} \big(h' * w_R\big)(E)\,\log\det S(E)\,dE.
$$
\end{theorem}

\begin{proof}
Integration by parts and substituting BK formula. \end{proof}

\section{Information Geometry and Born Probability}

\begin{theorem}[Born Probability = I-Projection]
\label{thm:born-iprojection}
For linear moment constraint $\mathcal{C}=\{p:\,\sum_i p_i a_i=b\}$ and reference $q$, minimal KL-divergence

$$
p^\star=\arg\min_{p\in\mathcal{C}}D_{\mathrm{KL}}(p\|q)
$$

has exponential family form $p^\star_i\propto q_i e^{\lambda a_i}$. If Born weights $w_i=\langle\psi,E_i\psi\rangle$ affinely expressible in constraint space, then $p^\star=w$ (Born probability).
\end{theorem}

\begin{proof}
Strict convexity of KL and Lagrange multipliers give exponential family and uniqueness; alignment condition derived from exponential family parameterization. POVM case by Naimark dilation to PVM then pushback. \end{proof}

\section{Pointer Basis and Ky Fan Minimum}

\begin{theorem}[Pointer Basis = Spectral Minimum]
\label{thm:pointer-basis}
For self-adjoint window operator $W_R$ and any $m$-dimensional orthogonal family $\{e_k\}$,

$$
\sum_{k=1}^m\langle e_k,W_Re_k\rangle\ge \sum_{k=1}^m\lambda_k^\uparrow(W_R),
$$

equality if and only if $\{e_k\}$ spans minimal eigensubspace of $W_R$ (Ky Fan minimum sum).
\end{theorem}

\begin{proof}
Standard Ky Fan variational principle (PNAS 1951). \end{proof}

\section{Non-Asymptotic Error Closure: NPE Decomposition}

\begin{theorem}[Nyquist--Poisson--EM Three-Term Decomposition]
\label{thm:npe-decomposition}
For energy-domain integral $I=\int_{\mathbb{R}} F(E)\,dE$ where $F=w_R\cdot(h\ast\rho_\star)$, under:
\begin{itemize}
\item Bandlimited: $\operatorname{supp}\widehat{F}\subset[-\Omega_F,\Omega_F]$;
\item Smoothness: $F\in C^{2M}(\mathbb{R})$, $F^{(2M)}\in L^1(\mathbb{R})$;
\item Sampling: step $\Delta>0$, truncation $|n|\le N$;
\end{itemize}

have discretization approximation

$$
I=\Delta\sum_{n=-N}^{N}F(n\Delta)+\underbrace{\varepsilon_{\mathrm{alias}}}_{\text{Poisson}}+\underbrace{R_{2M}}_{\text{EM}}+\underbrace{\varepsilon_{\mathrm{tail}}}_{\text{truncation}},
$$

where alias term $\varepsilon_{\mathrm{alias}}=0$ when $\Delta\le\pi/\Omega_F$ (Nyquist), EM remainder $|R_{2M}|\le \dfrac{2\zeta(2M)}{(2\pi)^{2M}}\int|F^{(2M)}|$, tail $|\varepsilon_{\mathrm{tail}}|\le\int_{|E|>N\Delta}|F|$.
\end{theorem}

\begin{proof}
Apply Poisson summation: for bandlimited $F$, replicas at $k\neq 0$ fall outside support when Nyquist satisfied. Apply $2M$-order Euler--Maclaurin to finite sum, obtaining Bernoulli corrections and explicit remainder. Tail from truncation. \end{proof}

\section{Multi-Window Frames and Wexler--Raz}

\begin{theorem}[Wexler--Raz Biorthogonality for Multi-Window]
\label{thm:wexler-raz}
For Gabor frame with time-frequency lattice $(\alpha,\beta)$ satisfying $\alpha\beta\le 1$, window $g$ and dual window $\widetilde{g}$ satisfy Wexler--Raz biorthogonality relation:

$$
\sum_{n\in\mathbb{Z}}g(t-n\alpha)\,\overline{\widetilde{g}(t-n\alpha)}\,e^{2\pi i m\beta t}=\frac{1}{\beta}\,\delta_{m,0},\quad \forall\,m\in\mathbb{Z},\text{ a.e. }t.
$$

Equivalently in frequency domain:

$$
\sum_{k\in\mathbb{Z}}\widehat{g}(\xi-k/\alpha)\,\overline{\widehat{\widetilde{g}}(\xi-k/\alpha)}=\alpha,\quad \text{a.e. }\xi.
$$
\end{theorem}

\begin{proof}
Standard result from Gabor analysis (Daubechies--Landau--Landau 1995). Follows from Poisson summation and frame operator properties. \end{proof}
